```javascript
  constructor(x, y, type) {

  }
}

//this is a specific class that extends the Movable class, so it can take advantage of all the properties that it inherits

  constructor(x,y) {
    super(x,y, 'Hero')
  }
}

  constructor(x,y) {

  }
}

//a hero can move...

//but a tree cannot

```
