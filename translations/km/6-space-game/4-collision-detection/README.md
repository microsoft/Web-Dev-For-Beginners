function intersectRect(r1, r2) {
  return !(r2.left > r1.right ||
    r2.right < r1.left ||
    r2.top > r1.bottom ||
    r2.bottom < r1.top);
}

бһ”бҹ’бһҡбһҹбһ·бһ“бһ”бһҫбһӮбҹ’бһҳбһ¶бһ“бһӣбһҖбҹ’бһҒбһҒбһҺбҹ’бһҢбһҺбһ¶бһҠбһјбһ…бһҒбһ¶бһ„бһӣбһҫбһҳбһҪбһҷбһҺбһ¶бһ“бҹ„бҹҮбһ‘бҹҒ бһҠбһ»бҹҶбһ–бһҸбҹӢбһ‘бһ¶бҹҶбһ„бһ–бһёбһҡбһҸбҹ’бһҡбһјбһңбһҳбһ¶бһ“бһҖбһ¶бһҡбһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…бһӮбҹ’бһ“бһ¶бҹ” бһңбһ·бһ’бһёбһ“бҹҒбҹҮбһҹбҹ’бһҡбһҠбҹҖбһ„бһ“бһ№бһ„бһҖбһ¶бһҡбһ”бһ»бһӮбҹ’бһӮбһӣбһ·бһҖбһҡбҹүбһ¶бһҢбһ¶бһҠбҹӮбһӣбһҖбҹҶбһҺбһҸбҹӢбһҗбһ¶бһҸбһҫбһҷбһ“бҹ’бһҸбһ бҹ„бҹҮбһ–бһёбһҡбһҳбһ¶бһ“бһ…бһҳбҹ’бһ„бһ¶бһҷбһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ¶бһ“бҹӢбһ‘бҹҒбһ‘бҹҒбҹ”

## бһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„бһңбһҠбҹ’бһҸбһңбһҸбҹ’бһҗбһ»

бһ–бҹҒбһӣбһҖбһ¶бҹҶбһ—бҹ’бһӣбһҫбһ„бһЎбһ¶бһҹбҹҠбҹӮбһҡбһҮбҹ’бһҡбһ»бҹҮбһӣбһҫбһҹбһҸбҹ’бһҡбһјбһң бһңбһҸбҹ’бһҗбһ»бһ‘бһ¶бҹҶбһ„бһ–бһёбһҡбһҸбҹ’бһҡбһјбһңбһ”бһ¶бһ“бһӣбһ»бһ”бһ…бҹҒбһүбһ–бһёбһ бҹ’бһӮбҹҒбһҳбҹ” бһ‘бҹ„бҹҮбһҮбһ¶бһҷбҹүбһ¶бһ„бһҺбһ¶ бһҖбһ¶бһҡбһӣбһ»бһ”бһңбһҸбҹ’бһҗбһ»бһҖбһҺбҹ’бһҸбһ¶бһӣбһңбһҠбҹ’бһҸбһўбһ¶бһ…бһ”бһҺбҹ’бһҸбһ¶бһӣбһІбҹ’бһҷбһ”бһүбҹ’бһ бһ¶бһ бһҪбһҹбһ”бҹ’бһҡбһҫ - бһҮбһ¶бһ”бһ‘бһ–бһ·бһҹбҹ„бһ’бһ“бҹҚбһ–бһёбһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’бһҖбһ»бҹҶбһ–бҹ’бһҷбһјбһ‘бҹҗбһҡбһҠбһјбһ… Apollo Guidance Computer бҹ” бһҮбҹҶбһ“бһҪбһҹ бһҷбһҫбһ„бһ”бҹ’бһҡбһҫбһңбһ·бһ’бһё "бһҹбһҳбҹ’бһӮбһ¶бһӣбҹӢбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһӣбһ»бһ”" бһҠбҹӮбһӣбһӣбһ»бһ”бһңбһҸбҹ’бһҗбһ»бһҠбҹ„бһҷбһҹбһ»бһңбһҸбҹ’бһҗбһ·бһ—бһ¶бһ–бһ“бҹ…бһ…бһ“бҹ’бһӣбҹ„бҹҮбһңбһҠбҹ’бһҸбҹ”

enemy.dead = true;

**бһ бҹҒбһҸбһ»бһўбҹ’бһңбһёбһ”бһ¶бһ“бһҮбһ¶бһңбһ·бһ’бһёбһ“бҹҒбҹҮбһ”бһ¶бһ“бһӣбҹ’бһў:**
- бһҷбһҫбһ„бһҹбһҳбҹ’бһӮбһ¶бһӣбҹӢбһңбһҸбҹ’бһҗбһ»бһҮбһ¶ "бһҹбҹ’бһӣбһ¶бһ”бҹӢ" бһ”бҹүбһ»бһ“бҹ’бһҸбҹӮбһҳбһ·бһ“бһӣбһ»бһ”бһңбһ¶бһ“бҹ„бҹҮбһ‘бҹҒ
- бһңбһ¶бһўбһ“бһ»бһүбҹ’бһүбһ¶бһҸбһІбҹ’бһҷбһңбһҠбҹ’бһҸбһ бҹ’бһӮбҹҒбһҳбһ”бһ…бҹ’бһ…бһ»бһ”бҹ’бһ”бһ“бҹ’бһ“бһ”бһүбҹ’бһ…бһ”бҹӢбһҠбҹ„бһҷбһҹбһ»бһңбһҸбҹ’бһҗбһ·бһ—бһ¶бһ–
- бһӮбҹ’бһҳбһ¶бһ“бһҖбһ¶бһҡбһ”бһүбҹ’бһ бһ¶бһ бһҪбһҹбһ”бҹ’бһҡбһҫ бһ–бһёбһҖбһ¶бһҡбһ”бҹ’бһҡбһҫбһңбһҸбҹ’бһҗбһ»бһҠбҹӮбһӣбһ”бһ¶бһ“бһ‘бҹ…бһҡбһҪбһ…бһ бһҫбһҷ!

бһ”бһ“бҹ’бһ‘бһ¶бһ”бҹӢбһҳбһҖбһ…бҹ’бһҡбһҖбһҷбһҖбһңбһҸбҹ’бһҗбһ»бһҠбҹӮбһӣбһҳбһ·бһ“бһҸбҹ’бһҡбһјбһңбһ”бһ¶бһ“бһҹбһҳбҹ’бһӮбһ¶бһӣбҹӢбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһӣбһ»бһ”бһҳбһ»бһ“бһҳбҹүбҹ„бһ„бһ”бһ„бҹ’бһ бһ¶бһүбһ”бһ“бҹ’бһ‘бһ¶бһ”бҹӢ:

```javascript
gameObjects = gameObjects.filter(go => !go.dead);
```

**бһҖбһ¶бһҡбһ…бҹҶбһ“бҹҒбһүбһ“бҹҒбҹҮбһ’бҹ’бһңбһҫбһҠбһјбһ…бһҮбһ¶:**
- бһ”бһ„бҹ’бһҖбһҫбһҸбһ”бһүбҹ’бһҮбһёбһҗбҹ’бһҳбһёбһҳбһ¶бһ“бһҸбҹӮбһ“бҹ…бһңбһҸбҹ’бһҗбһ» "бһҡбһҹбҹӢ"
- бһ”бһ¶бһҸбҹӢбһ”бһ„бҹӢбһңбһҸбҹ’бһҗбһ»бһҠбҹӮбһӣбһҹбһҳбҹ’бһӮбһ¶бһӣбҹӢбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһӣбһ»бһ”
- бһҡбһҖбҹ’бһҹбһ¶бһ бҹ’бһӮбҹҒбһҳбһҡбһ”бһҹбҹӢбһўбҹ’бһ“бһҖбһІбҹ’бһҷбһҡбһӣбһјбһ“
- бһҖбһ¶бһҡбһ–бһ¶бһҡбһ”бһүбҹ’бһ бһ¶бһҹбҹ’бһҸбһ»бһҖбһҳбҹүбҹҒбһҳбҹүбһјбһҡбһёбһ–бһёбһңбһҸбҹ’бһҗбһ»бһҠбҹӮбһӣбһҒбһјбһ…бһҒбһ¶бһҸ бһ“бһ·бһ„бһ…бҹҶбһҺбһ»бҹҮбһҮбһ¶бһ…бҹ’бһҡбһҫбһ“

## бһўбһ“бһ»бһңбһҸбҹ’бһҸбһҳбҹҒбһҖбһ¶бһ“бһ·бһ…бһЎбһ¶бһҹбҹҠбҹӮбһҡ

бһӮбҹ’бһҡбһ¶бһ”бҹӢбһҖбһ¶бҹҶбһ—бҹ’бһӣбһҫбһ„бһЎбһ¶бһҹбҹҠбҹӮбһҡ бһҖбҹ’бһ“бһ»бһ„бһ бҹ’бһӮбҹҒбһҳбһҠбһҫбһҡбһҸбһ¶бһҳбһӮбҹ„бһӣбһҖбһ¶бһҡбһҺбҹҚбһҠбһјбһ…бһ‘бһјбһҸбһ—бһёбһҸбһ»бһ„бһҖбҹ’бһ“бһ»бһ„ Star Trek - бһңбһ¶бһҮбһ¶бһңбһҸбҹ’бһҗбһ»бһ”бһ¶бһҸбҹӢбһ”бһ„бҹӢбһҠбҹӮбһӣбһ’бҹ’бһңбһҫбһҠбҹҶбһҺбһҫбһҡбһҮбһ¶бһӣбҹҶбһ“бһ¶бҹҶбһҸбҹ’бһҡбһ„бҹӢбһҡбһ бһјбһҸбһҠбһӣбҹӢбһңбһ¶бһ”бҹүбҹҮбһ—бһ¶бһӮбһёбһҺбһ¶бһҳбһҪбһҷбҹ” бһҡбһ¶бһӣбҹӢбһҖбһ¶бһҡбһ…бһ»бһ… spacebar бһ”бһ„бҹ’бһҖбһҫбһҸбһңбһҸбҹ’бһҗбһ»бһӣбһ¶бһҹбҹҠбҹӮбһҡ бһҗбҹ’бһҳбһёбһҳбһҪбһҷ бһҠбҹӮбһӣбһ•бҹ’бһӣбһ¶бһҹбҹӢбһ‘бһёбһӣбһҫбһўбҹҒбһҖбҹ’бһҡбһ„бҹӢбҹ”

бһҠбһҫбһҳбҹ’бһ”бһёбһІбҹ’бһҷбһңбһ¶бһҠбҹҶбһҺбһҫбһҡбһҖбһ¶бһҡ бһҷбһҫбһ„бһҸбҹ’бһҡбһјбһңбһҹбһҳбҹ’бһҡбһ”бһҹбһҳбҹ’бһҡбһҪбһӣбһ”бҹүбһ»бһ“бҹ’бһҳбһ¶бһ“бһ•бҹ’бһ“бҹӮбһҖбһ•бҹ’бһҹбҹҒбһ„бҹ—бҹ–

**бһ”бһ“бҹ’бһ‘бһ¶бһ”бҹӢбһ”бһ“бҹ’бһҹбҹҶбһӮбһҪбһҡбһҡбҹҖбһ“:**
- **бһ”бһ„бҹ’бһҖбһҫбһҸ** бһўбҹ’бһ“бһҖбһҸбҹҶбһҺбһ¶бһ„бһЎбһ¶бһҹбҹҠбҹӮбһҡ бһҠбҹӮбһӣбһ•бһӣбһ·бһҸбһ–бһёбһ‘бһёбһҸбһ¶бҹҶбһ„бһңбһёбһҡбһ”бһ»бһҡбһҹ
- **бһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„** бһҖбһ¶бһҡбһ”бһүбҹ’бһ…бһјбһӣбһҖбҹ’бһҸбһ¶бһҡбһ…бһ»бһ… бһҠбһҫбһҳбҹ’бһ”бһёбһ”бһ„бҹ’бһҖбһҫбһҸбһЎбһ¶бһҹбҹҠбҹӮбһҡ
- **бһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„** бһ…бһӣбһ“бһ¶ бһ“бһ·бһ„бһңбһҠбҹ’бһҸбһҮбһёбһңбһ·бһҸбһЎбһ¶бһҹбҹҠбҹӮбһҡ
- **бһ”бһ„бҹ’бһҖбһҫбһҸ** бһҖбһ¶бһҡбһ–бһ·бһ–бһҺбҹҢбһ“бһ¶бһ—бҹ’бһ“бҹӮбһҖбһ„бһ¶бһҡбһҡбһ”бһҹбҹӢбһӮбҹ’бһҡбһ¶бһ”бҹӢбһ”бҹӮбһҖбһЎбһ¶бһҹбҹҠбҹӮбһҡ

## бһўбһ“бһ»бһңбһҸбҹ’бһҸбһҖбһ¶бһҡбһҸбҹ’бһҡбһҪбһҸбһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһўбһҸбҹ’бһҡбһ¶бһңбһ¶бһҷбһ”бһ¶бһүбҹӢ

бһўбһҸбҹ’бһҡбһ¶бһңбһ¶бһҷбһ”бһ¶бһүбҹӢбһҳбһ·бһ“бһҳбһ¶бһ“бһҖбҹҶбһҺбһҸбҹӢбһ“бһ№бһ„бһ‘бҹҶбһ“бһ¶бһҖбҹӢбһ‘бҹҶбһ“бһ„бһҳбҹүбһ¶бһҹбҹҠбһёбһ“бһ бҹ’бһӮбҹҒбһҳ бһ бһҫбһҷбһ’бҹ’бһңбһҫбһІбҹ’бһҷбһ бҹ’бһӮбҹҒбһҳбһ„бһ¶бһҷбһ–бҹҒбһҖбһҖбҹ’бһ“бһ»бһ„бһҖбһ¶бһҡбһӣбҹҒбһ„бҹ” бһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’бһўбһ¶бһңбһ»бһ’бһ–бһ·бһҸбһ”бҹ’бһҡбһ¶бһҖбһҠбһҖбҹҸбһҳбһ¶бһ“бһҖбҹҶбһҺбһҸбҹӢбһҠбһјбһ…бһӮбҹ’бһ“бһ¶ - бһ”бҹӮбһ”бһ•бҹӮбһ“бһҖбһ¶бҹҶбһ—бҹ’бһӣбһҫбһ„ phaser бһ“бҹғ USS Enterprise бһҸбҹ’бһҡбһјбһңбһ–бҹҒбһӣбһҶбһҖбҹӢбһҗбҹ’бһҳбһҳбҹ’бһҠбһ„бһҳбҹ’бһҖбһ¶бһӣбҹ”

бһҷбһҫбһ„бһ“бһ№бһ„бһўбһ“бһ»бһңбһҸбҹ’бһҸбһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’ cooldown бһҠбҹӮбһӣбһ”бһүбҹ’бһҲбһ”бҹӢбһҖбһ¶бһҡбһ”бһ¶бһүбҹӢбһҠбҹҸбһҡбһ бҹҗбһҹбһҡбһ бһҪбһ“ бһҖбҹ’бһ“бһ»бһ„бһҒбһҺбҹҲбһҠбҹӮбһӣбһҡбһҖбҹ’бһҹбһ¶бһҖбһ¶бһҡбһҸбҹ’бһҡбһҪбһҸбһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһҶбҹ’бһӣбһҫбһҷбһҸбһ”бһ”бһ¶бһ“бҹ–

```javascript
class Cooldown {
  constructor(time) {
    this.cool = false;
    setTimeout(() => {
      this.cool = true;
    }, time);
  }
}

class Weapon {
  constructor() {
    this.cooldown = null;
  }
  
  fire() {
    if (!this.cooldown || this.cooldown.cool) {
      // бһ”бһ„бҹ’бһҖбһҫбһҸбһӮбҹ’бһҡбһ¶бһ”бҹӢбһ–бһ“бҹ’бһӣбһәбһЎбһ¶бһҹбҹҠбҹӮбһҡ
      this.cooldown = new Cooldown(500);
    } else {
      // бһўбһ¶бһңбһ»бһ’бһҖбҹҶбһ–бһ»бһ„бһҸбҹ’бһҡбһҮбһ¶бһҖбҹӢбһ‘бҹ…бһңбһ·бһү
    }
  }
}

**бһҡбһ”бҹҖбһ” cooldown бһҠбҹҶбһҺбһҫбһҡбһҖбһ¶бһҡ:**
- бһ“бҹ…бһ–бҹҒбһӣбһ”бһ„бҹ’бһҖбһҫбһҸ бһҹбҹҶбһ–бһ¶бһ’бһҹбһүбҹ’бһҮбһ¶бһҸбһ· "бһҖбҹ’бһҸбҹ…" (бһҳбһ·бһ“бһўбһ¶бһ…бһ”бһ¶бһүбҹӢбһ”бһ¶бһ“бһ‘бҹҒ)
- бһ”бһ“бҹ’бһ‘бһ¶бһ”бҹӢбһ–бһёбһ–бҹҒбһӣбһңбҹҒбһӣбһ¶бһ”бһүбҹ’бһ…бһ”бҹӢ бһңбһ¶бһ”бҹ’бһҡбҹӮбһ‘бҹ… "бһҸбҹ’бһҡбһҮбһ¶бһҖбҹӢ" (бһҡбһҪбһ…бһҡбһ¶бһӣбҹӢбһ”бһ¶бһүбҹӢ)
- бһҳбһ»бһ“бһ–бҹҒбһӣбһ”бһ¶бһүбҹӢ бһҷбһҫбһ„бһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһҗбһ¶: "бһўбһ¶бһңбһ»бһ’бһҸбҹ’бһҡбһҮбһ¶бһҖбҹӢбһ‘бҹҒ?"
- бһңбһ¶бһ”бһүбҹ’бһҲбһ”бҹӢбһҖбһ¶бһҡбһ…бһ»бһ…бһҸбҹӮбһ…бҹ’бһҡбһҫбһ“бһ”бһ“бҹ’бһ‘бһ¶бһ“бҹӢ бһ”бҹүбһ»бһ“бҹ’бһҸбҹӮбһҖбҹҸбһҡбһҖбҹ’бһҹбһ¶бһҖбһ¶бһҡбһҸбҹ’бһҡбһҪбһҸбһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһҶбҹ’бһӣбһҫбһҷбһҸбһ”бһ”бһ¶бһ“

вң… бһўбһ“бһ»бһңбһҸбҹ’бһҸбһ‘бҹ…бһҳбҹҒбһҡбҹҖбһ“бһ‘бһё 1 бһҖбҹ’бһ“бһ»бһ„бһ”бһҺбҹ’бһҠбһ»бҹҶбһ бҹ’бһӮбҹҒбһҳбһўбһ¶бһҖбһ¶бһҹ бһҠбһҫбһҳбҹ’бһ”бһёбһ…бһ„бһ…бһ¶бҹҶбһўбҹҶбһ–бһё cooldownsбҹ”

## бһҡбҹҖбһ”бһ…бҹҶбһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’бһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…

бһўбҹ’бһ“бһҖбһ“бһ№бһ„бһ–бһ„бҹ’бһҡбһёбһҖбһҖбһјбһҠбһ бҹ’бһӮбҹҒбһҳбһўбһ¶бһҖбһ¶бһҹбһҠбҹӮбһӣбһҳбһ¶бһ“бһҡбһҪбһ…бһҮбһ¶бһҳбһҪбһҷбһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’бһҹбһҳбҹ’бһӮбһ¶бһӣбҹӢбһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…бҹ” бһҠбһјбһ…бһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’бһҮбҹҖбһҹбһңбһ¶бһ„бһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…бһҹбҹ’бһңбҹҗбһҷбһ”бҹ’бһҡбһңбһҸбҹ’бһҸбһ·вҖӢбһ“бҹғбһҹбҹ’бһҗбһ¶бһ“бһёбһҷбһўбһ¶бһҖбһ¶бһҹбһўбһ“бҹ’бһҸбһҡбһҮбһ¶бһҸбһ· бһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһ бҹ’бһӮбҹҒбһҳбһҡбһ”бһҹбҹӢбһўбҹ’бһ“бһҖ бһ“бһ№бһ„бһҸбҹ’бһҡбһҪбһҸбһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһ‘бһёбһҸбһ¶бҹҶбһ„бһңбһҸбҹ’бһҗбһ»бһҮбһ¶бһ”бһ“бҹ’бһҸбһ”бһ“бҹ’бһ‘бһ¶бһ”бҹӢ бһ бһҫбһҷбһҶбҹ’бһӣбһҫбһҷбһҸбһ”бһ–бҹҒбһӣбһҳбһ¶бһ“бһҖбһ¶бһҡбһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…бҹ”

бһ…бһ¶бһ”бҹӢбһ•бҹ’бһҸбһҫбһҳбһ–бһёбһҖбһјбһҠбһҳбҹҒбһҡбҹҖбһ“бһҳбһ»бһ“ бһўбҹ’бһ“бһҖбһ“бһ№бһ„бһ”бһ“бҹ’бһҗбҹӮбһҳбһҖбһ¶бһҡбһҹбһҳбҹ’бһӮбһ¶бһӣбҹӢбһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…бһҮбһ¶бһҳбһҪбһҷбһ“бһ№бһ„бһ…бҹ’бһ”бһ¶бһ”бҹӢбһ”бһүбҹ’бһҮбһ¶бһҖбҹӢбһҠбҹӮбһӣбһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„бһҖбһ¶бһҡбһ•бҹ’бһӮбһҪбһ•бҹ’бһӮбһ„бһңбһҸбҹ’бһҗбһ»бҹ”

> рҹ’Ў **бһ…бҹҶбһҺбһ¶бһҸбҹӢбһҗбҹ’бһ“бһ¶бһҖбҹӢбһҮбҹҶбһ“бһ¶бһү**: бһҡбһјбһ”бһҸбҹҶбһҺбһ¶бһ„бһЎбһ¶бһҹбҹҠбҹӮбһҡ бһҳбһ¶бһ“бһҡбһҪбһ…бһҖбҹ’бһ“бһ»бһ„бһҗбһҸ assets бһҡбһ”бһҹбҹӢбһўбҹ’бһ“бһҖбһ бһҫбһҷбһ”бһ¶бһ“бһҷбҹ„бһ„бһҖбҹ’бһ“бһ»бһ„бһҖбһјбһҠбһҡбһ”бһҹбҹӢбһўбҹ’бһ“бһҖ бһҡбҹҖбһ”бһ…бҹҶбһҡбһҪбһ…бһ бһҫбһҷбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһўбһ“бһ»бһңбһҸбҹ’бһҸбҹ”

### бһ…бҹ’бһ”бһ¶бһ”бҹӢбһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…бһҠбҹӮбһӣбһҸбҹ’бһҡбһјбһңбһўбһ“бһ»бһңбһҸбҹ’бһҸ

**бһҳбҹҒбһҖбһ¶бһ“бһ·бһҖбһ бҹ’бһӮбҹҒбһҳбһҠбҹӮбһӣбһҸбҹ’бһҡбһјбһңбһ”бһ“бҹ’бһҗбҹӮбһҳ:**
1. **бһЎбһ¶бһҹбҹҠбҹӮбһҡвҖӢбһ•бҹ…бһҹбһҸбҹ’бһҡбһјбһң**: бһңбһҸбҹ’бһҗбһ»бһҹбһҸбҹ’бһҡбһјбһңбһҸбҹ’бһҡбһјбһңбһ”бһ¶бһ“бһ”бҹҶбһ”бҹӮбһҖбһ“бҹ…бһ–бҹҒбһӣбһ”бһ¶бһүбҹӢбһҠбҹ„бһҷбһӮбҹ’бһҡбһ¶бһ”бҹӢбһЎбһ¶бһҹбҹҠбҹӮбһҡ
2. **бһЎбһ¶бһҹбҹҠбҹӮбһҡвҖӢбһ•бҹ…бһҠбҹӮбһ“бһўбҹҒбһҖбҹ’бһҡбһ„бҹӢ**: бһҠбһ»бһҸбһҖбһ¶бҹҶбһ—бҹ’бһӣбһҫбһ„бһЎбһ¶бһҹбҹҠбҹӮбһҡ бһҸбҹ’бһҡбһјбһңбһ”бһ¶бһ“бһҠбһҖбһ…бҹҒбһү бһ“бҹ…бһ–бҹҒбһӣбһ–бһҪбһҖбһңбһ¶бһ‘бҹ…бһҠбһӣбҹӢбһҖбҹҶбһ–бһјбһӣбһўбҹҒбһҖбҹ’бһҡбһ„бҹӢ
3. **бһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…бһҹбһҸбҹ’бһҡбһјбһң бһ“бһ·бһ„бһңбһёбһҡбһ”бһ»бһҡбһҹ**: бһңбһҸбҹ’бһҗбһ»бһ‘бһ¶бҹҶбһ„бһ–бһёбһҡбһҸбҹ’бһҡбһјбһңбһ”бҹҶбһ”бҹӮбһҖ бһ“бҹ…бһ–бҹҒбһӣбһ”бҹүбҹҮбһӮбҹ’бһ“бһ¶
4. **бһҹбһҸбҹ’бһҡбһјбһңбһҳбһҖбһҠбһӣбҹӢбһ•бҹ’бһ‘бҹғбһҖбҹ’бһҡбҹ„бһҳ**: бһҗбҹ’бһ„бҹғбһ”бһүбҹ’бһ…бһ”бҹӢбһ бҹ’бһӮбҹҒбһҳбһ“бҹ…бһ–бҹҒбһӣбһҹбһҸбҹ’бһҡбһјбһңбһҠбһӣбҹӢбһ•бҹ’бһ‘бҹғбһҖбҹ’бһҡбҹ„бһҳбһўбҹҒбһҖбҹ’бһҡбһ„бҹӢ

### рҹ”„ **бһҖбһ¶бһҡбһҸбҹ’бһҡбһҪбһҸбһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһ•бҹ’бһ“бҹӮбһҖбһ”бһ„бҹ’бһҡбҹҖбһ“**
**бһҳбһјбһӣбһҠбҹ’бһӢбһ¶бһ“бһҖбһ¶бһҡбһҡбһҖбһҳбһҫбһӣбһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…**бҹ– бһҳбһ»бһ“бһўбһ“бһ»бһңбһҸбҹ’бһҸ бһ”бҹ’бһҡбһ¶бһҖбһҠбһҗбһ¶бһўбҹ’бһ“бһҖбһҷбһӣбҹӢбһ–бһёбҹ–

бһҮбһ¶бһҳбһ»бһ“бһҹбһ·бһ“ бһҷбһҫбһ„бһ…бһ„бҹӢбһ”бһ„бҹ’бһҡбҹҖбһ“бһІбҹ’бһҷбһңбһҸбҹ’бһҗбһ»бһ бҹ’бһӮбҹҒбһҳбһўбһ’бһ·бһ”бҹ’бһ”бһ¶бһҷбһ–бһёбһҠбҹӮбһ“бһҖбҹҶбһҺбһҸбҹӢбһҡбһ”бһҹбҹӢбһңбһ¶бҹ” бһ”бһ“бҹ’бһҗбҹӮбһҳбһңбһ·бһ’бһёбһҹбһ¶бһҹбҹ’бһҡбҹ’бһҸбһ“бҹҒбҹҮбһ‘бҹ…бһҗбҹ’бһ“бһ¶бһҖбҹӢ `GameObject` бһҡбһ”бһҹбҹӢбһўбҹ’бһ“бһҖбҹ–

#### 2. бһўбһ“бһ»бһңбһҸбҹ’бһҸбһҖбһ¶бһҡбһҹбһҳбҹ’бһӮбһ¶бһӣбҹӢбһҖбһ¶бһҡбһ”бҹүбҹҮбһ–бһ¶бһӣбҹӢ

бһҘбһЎбһјбһңбһ“бҹҒбҹҮ бһўбҹ’бһ“бһҖбһҸбҹ’бһҡбһјбһңбһ”бһ„бҹ’бһҖбһҫбһҸбһұбҹ’бһҷбһҳбһ¶бһ“бһҳбһјбһӣбһҠбҹ’бһӢбһ¶бһ“бһҹбһҳбҹ’бһӮбһ¶бһӣбҹӢбһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ… - бһҳбһ»бһҒбһ„бһ¶бһҡвҖӢбһҠбҹӮбһӣбһҹбҹ’бһӮбһ¶бһӣбҹӢбһ–бҹҒбһӣбһҜбһҖбһҸбһ¶бһҠбһ»бҹҶбһ–бһҸбҹӢбһ–бһёбһҡбһ”бҹүбҹҮбһ‘бҹҶбһ“бһ¶бһҖбҹӢбһ‘бҹҶбһ“бһ„бһӮбҹ’бһ“бһ¶бҹ–

function intersectRect(r1, r2) {
  return !(
    r2.left > r1.right ||
    r2.right < r1.left ||
    r2.top > r1.bottom ||
    r2.bottom < r1.top
  );
}
```

- **бһҸбҹ’бһҡбһЎбһ”** `false` бһ”бҹ’бһҡбһҹбһ·бһ“бһ”бһҫбһӣбһҖбҹ’бһҒбһҒбһҺбҹ’бһҢбһ”бҹҶбһ”бҹӮбһҖбһҺбһ¶бһҳбһҪбһҷбһҳбһ¶бһ“бһҹбһ»бһ–бһӣбһ—бһ¶бһ–
- **бһ”бһүбҹ’бһҮбһ¶бһҖбҹӢ** бһҖбһҡбһҺбһёбһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ… бһ–бҹҒбһӣбһӮбҹ’бһҳбһ¶бһ“бһҖбһ¶бһҡбһ”бҹҶбһ”бҹӮбһҖ
- **бһ”бҹ’бһҡбһҫ** бһҷбһ»бһ‘бҹ’бһ’бһҹбһ¶бһҹбҹ’бһҸбҹ’бһҡбһ”бһҠбһ·бһҹбҹҒбһ’бһҹбҹҶбһҺбһҪбһҡ бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҖбһ¶бһҡбһҡбһҖбһҠбҹҶбһҺбһҫбһҡбһҖбһ¶бһҡбһӣбһҝбһ“

- **бһ’бһ¶бһ“бһ¶** бһҖбһ¶бһҡбһҹбһҳбҹ’бһўбһ¶бһҸбһҸбҹ’бһҡбһ№бһҳбһҸбҹ’бһҡбһјбһңбһ”бһ“бҹ’бһ‘бһ¶бһ”бҹӢбһ–бһёбһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…

#### 4. бһ”бһ„бҹ’бһҖбһҫбһҸбһҗбҹ’бһ“бһ¶бһҖбҹӢ Laser

бһўбһ“бһ»бһңбһҸбҹ’бһҸбһӮбҹ’бһҡбһ¶бһ”бҹӢбһ”бһ¶бһүбҹӢбһЎбһ¶бһҹбҹҠбҹӮбһҡ бһҠбҹӮбһӣбһ•бҹ’бһӣбһ¶бһҹбҹӢбһ‘бһёбһ‘бҹ…бһӣбһҫ бһ“бһ·бһ„бһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„бһңбһҠбҹ’бһҸбһҮбһёбһңбһ·бһҸбһҡбһ”бһҹбҹӢбһңбһ¶бҹ–

```javascript
class Laser extends GameObject {
  constructor(x, y) {
    super(x, y);
    this.width = 9;
    this.height = 33;
    this.type = 'Laser';
    this.img = laserImg;
    
    let id = setInterval(() => {
      if (this.y > 0) {
        this.y -= 15;
      } else {
        this.dead = true;
        clearInterval(id);
      }
    }, 100);
  }
}
```

**бһҖбһ¶бһҡбһўбһ“бһ»бһңбһҸбҹ’бһҸбһҗбҹ’бһ“бһ¶бһҖбҹӢбһ“бҹҒбҹҮ:**
- **бһ”бһ“бҹ’бһҸ** бһ–бһёбһҗбҹ’бһ“бһ¶бһҖбҹӢ GameObject бһҠбһҫбһҳбҹ’бһ”бһёбһ‘бһ‘бһҪбһӣбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһҳбһјбһӣбһҠбҹ’бһӢбһ¶бһ“
- **бһҖбҹҶбһҺбһҸбҹӢ** бһ‘бҹҶбһ бҹҶбһҹбһҳбһҡбһҳбҹ’бһҷбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҡбһјбһ”бһҸбҹҶбһҺбһ¶бһ„бһЎбһ¶бһҹбҹҠбҹӮбһҡ
- **бһ”бһ„бҹ’бһҖбһҫбһҸ** бһ…бһӣбһ“бһ¶бһЎбһҫбһ„бһҮбһ¶бһ”бҹ’бһҡбһ…бһ¶бҹҶбһҠбҹ„бһҷбһ”бҹ’бһҡбһҫ `setInterval()`
- **бһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„** бһҖбһ¶бһҡбһҡбҹҶбһӣбһ¶бһҷбһҒбҹ’бһӣбһҪбһ“бһҜбһ„бһ“бҹ…бһ–бҹҒбһӣбһ‘бҹ…бһҠбһӣбҹӢбһҖбҹҶбһ–бһјбһӣбһўбҹҒбһҖбҹ’бһҡбһ„бҹӢ
- **бһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„** бһ–бҹҒбһӣбһңбҹҒбһӣбһ¶бһўбһ“бһ»бһңбһҸбҹ’бһҸбһ“бҹҚбһ“бһ·бһ„бһҖбһ¶бһҡбһҹбһҳбҹ’бһўбһ¶бһҸ

#### 5. бһўбһ“бһ»бһңбһҸбҹ’бһҸбһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’бһҹбһҳбҹ’бһӮбһ¶бһӣбҹӢбһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…

бһ”бһ„бҹ’бһҖбһҫбһҸбһҳбһ»бһҒбһ„бһ¶бһҡбһҹбһҳбҹ’бһӮбһ¶бһӣбҹӢбһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…бһҹбһҡбһ»бһ”бҹ–

```javascript
function updateGameObjects() {
  const enemies = gameObjects.filter(go => go.type === 'Enemy');
  const lasers = gameObjects.filter(go => go.type === "Laser");
  
  // бһҹбһ¶бһҖбһӣбҹ’бһ”бһ„бһҖбһ¶бһҡбһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…бһҡбһңбһ¶бһ„бһ”бһ¶бһүбҹӢбһңбһ¶бһҷбһҹбһҸбҹ’бһҡбһјбһң
  lasers.forEach((laser) => {
    enemies.forEach((enemy) => {
      if (intersectRect(laser.rectFromGameObject(), enemy.rectFromGameObject())) {
        eventEmitter.emit(Messages.COLLISION_ENEMY_LASER, {
          first: laser,
          second: enemy,
        });
      }
    });
  });

  // бһ бҹӮбһҖбһңбһҸбҹ’бһҗбһ»бһҠбҹӮбһӣбһ”бһ¶бһ“бһ”бҹҶбһ•бҹ’бһӣбһ¶бһү
  gameObjects = gameObjects.filter(go => !go.dead);
}
```

**бһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’бһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…бһ“бҹҒбҹҮ:**
- **бһҸбҹ’бһҡбһ„бҹӢ** бһңбһҸбҹ’бһҗбһ»бһ бҹ’бһӮбҹҒбһҳбһҸбһ¶бһҳбһ”бҹ’бһҡбһ—бҹҒбһ‘ бһҠбһҫбһҳбҹ’бһ”бһёбһ’бҹ’бһңбһҫбһҸбҹҒбһҹбҹ’бһҸбһ”бһ¶бһ“бһӣбһҝбһ“
- **бһҸбҹҒбһҹбҹ’бһҸ** бһӮбҹ’бһҡбһ”бҹӢбһЎбһ¶бһҹбҹҠбҹӮбһҡ бһ‘бһӣбҹӢбһ“бһ№бһ„бһҹбһҸбҹ’бһҡбһјбһңбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһ…бҹҶбһҺбһ»бһ…бһҶбҹ’бһӣбһ„
- **бһ”бһүбҹ’бһ…бҹҒбһү** бһ–бҹ’бһҡбһ№бһҸбҹ’бһҸбһ·бһҖбһ¶бһҡбһҺбҹҚбһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ… бһ“бҹ…бһ–бҹҒбһӣбһҡбһҖбһғбһҫбһүбһҖбһ¶бһҡвҖӢбһҶбҹ’бһӣбһ„
- **бһҹбһҳбҹ’бһўбһ¶бһҸ** бһңбһҸбҹ’бһҗбһ»бһҠбҹӮбһӣбһ”бҹҶбһ•бҹ’бһӣбһ¶бһүбһ”бһ“бҹ’бһ‘бһ¶бһ”бҹӢбһ–бһёбһҠбҹҶбһҺбһҫбһҡбһҖбһ¶бһҡбһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…

> вҡ пёҸ **бһҹбҹҶбһҒбһ¶бһ“бҹӢ**бҹ– бһ”бһ“бҹ’бһҗбҹӮбһҳ `updateGameObjects()` бһ‘бҹ…бһңбһҠбҹ’бһҸбһ бҹ’бһӮбҹҒбһҳбһҹбҹҶбһҒбһ¶бһ“бҹӢбһ“бҹ…бһҖбҹ’бһ“бһ»бһ„ `window.onload` бһҠбһҫбһҳбҹ’бһ”бһёбһ”бһҫбһҖбһ…бҹҶбһ бһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’бһҹбһҳбҹ’бһӮбһ¶бһӣбҹӢбһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…бҹ”

#### 6. бһ”бһ“бҹ’бһҗбҹӮбһҳбһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’ cooldown бһ‘бҹ…бһҗбҹ’бһ“бһ¶бһҖбҹӢ Hero

бһ”бһ„бҹ’бһҖбһҫбһ“бһҗбҹ’бһ“бһ¶бһҖбҹӢ Hero бһҮбһ¶бһҳбһҪбһҷбһҳбҹҒбһҖбһ¶бһ“бһ·бһ…бһҖбһ¶бһҡбһ”бһ¶бһүбҹӢ бһ“бһ·бһ„бһҖбһ¶бһҡбһҖбҹҶбһҺбһҸбҹӢбһўбһҸбҹ’бһҡбһ¶бҹ–

```javascript
class Hero extends GameObject {
  constructor(x, y) {
    super(x, y);
    this.width = 99;
    this.height = 75;
    this.type = "Hero";
    this.speed = { x: 0, y: 0 };
    this.cooldown = 0;
  }
  
  fire() {
    gameObjects.push(new Laser(this.x + 45, this.y - 10));
    this.cooldown = 500;

    let id = setInterval(() => {
      if (this.cooldown > 0) {
        this.cooldown -= 100;
      } else {
        clearInterval(id);
      }
    }, 200);
  }
  
  canFire() {
    return this.cooldown === 0;
  }
}
```

**бһҖбһ¶бһҡбһҷбһӣбҹӢбһ–бһёбһҗбҹ’бһ“бһ¶бһҖбҹӢ Hero бһ”бһ¶бһ“бһ”бһ„бҹ’бһҖбһҫбһ“:**
- **бһ…бһ¶бһ”бҹӢбһ•бҹ’бһҠбһҫбһҳ** бһҳбҹүбҹ„бһ„ cooldown бһҮбһ¶бһҹбһјбһ“бҹ’бһҷ (бһҡбһҪбһ…бһҡбһ¶бһӣбҹӢбһ”бһ¶бһүбҹӢ)
- **бһ”бһ„бҹ’бһҖбһҫбһҸ** бһңбһҸбҹ’бһҗбһ»бһӣбһ¶бһҹбҹҠбҹӮбһҡ бһҠбҹӮбһӣбһҳбһ¶бһ“бһ‘бһёбһҸбһ¶бҹҶбһ„бһӣбһҫбһ‘бһјбһҖбһңбһёбһҡбһ”бһ»бһҡбһҹ
- **бһҖбҹҶбһҺбһҸбҹӢ** бһ–бҹҒбһӣ cooldown бһҠбһҫбһҳбҹ’бһ”бһёбһ‘бһ”бҹӢбһҹбҹ’бһҖбһ¶бһҸбҹӢбһҖбһ¶бһҡбһ”бһ¶бһүбҹӢбһЎбһҫбһ„бһӣбһҝбһ“бһ–бҹҒбһҖ
- **бһҖбһ¶бһҸбҹӢбһ”бһ“бҹ’бһҗбһҷ** бһҳбҹүбҹ„бһ„ cooldown бһҠбҹ„бһҷбһҖбһ¶бһҡбһўбһ¶бһ”бҹӢбһҠбҹҒбһҸбһ”бҹ’бһҡбһҖбҹ’бһҡбһҸбһё
- **бһ•бҹ’бһҠбһӣбҹӢ** бһҳбһ»бһҒбһ„бһ¶бһҡ `canFire()` бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһҹбҹ’бһҗбһ¶бһ“бһ—бһ¶бһ–бһ”бһ¶бһүбҹӢ

### рҹ”„ **бһҖбһ¶бһҡбһҸбҹ’бһҡбһҪбһҸбһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһ•бҹ’бһ“бҹӮбһҖбһ”бһ„бҹ’бһҡбҹҖбһ“**
**бһҖбһ¶бһҡбһҷбһӣбҹӢбһҠбһ№бһ„бһҹбһҡбһ»бһ”**бҹ– бһ”бҹ’бһҡбһ¶бһҖбһҠбһҗбһ¶бһўбҹ’бһ“бһҖбһ…бҹҒбҹҮбһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„бһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’бһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ… бҹ–
- вң… бһҡбһ”бҹҖбһ”бһҠбҹӮбһ“бһҖбҹҶбһҺбһҸбҹӢбһҠбһ»бҹҶбһ–бһҸбҹӢбһҮбһҪбһҷбһҖбһ¶бһҡбһҹбһҳбҹ’бһӮбһ¶бһӣбҹӢбһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…бһ”бһ¶бһ“бһӣбһҝбһ“?
- вң… бһ бҹҒбһҸбһ»бһўбҹ’бһңбһёбһҖбһ¶бһҡбһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„бһңбһҠбҹ’бһҸбһңбһҸбҹ’бһҗбһ»бһҳбһ¶бһ“бһҹбһ¶бһҡбҹҲбһҹбҹҶбһҒбһ¶бһ“бҹӢбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҹбҹ’бһҗбһ·бһҡбһ—бһ¶бһ–бһ бҹ’бһӮбҹҒбһҳ?
- вң… бһҳбһ»бһҒбһ„бһ¶бһҡ cooldown бһ–бҹ’бһҷбһ¶бһ”бһ¶бһӣбһ”бһүбҹ’бһ бһ¶бһ”бҹ’бһҡбһҹбһ·бһ‘бҹ’бһ’бһ—бһ¶бһ–бһҷбҹүбһ¶бһ„бһҠбһјбһ…бһҳбҹ’бһҸбҹҒбһ…?
- вң… бһҸбһҫбһҹбҹ’бһҗбһ¶бһ”бһҸбҹ’бһҷбһҖбһҳбҹ’бһҳбһ’бҹ’бһңбһҫбһҳбҹүбҹҒбһ… бһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„бһҖбһ¶бһҡбһҶбҹ’бһӣбһҫбһҷбһҸбһ”бһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…?

**бһҖбһ¶бһҡбһҹбһҳбҹ’бһҡбһ”бһҹбһҳбҹ’бһҡбһҪбһӣбһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’**бҹ– бһўбҹ’бһ“бһҖбһўбһ¶бһ…бһ”бһ„бҹ’бһ бһ¶бһүбҹ–
- **бһ—бһ¶бһ–бһҸбҹ’бһҡбһ№бһҳбһҸбҹ’бһҡбһјбһңбһӮбһҺбһ·бһҸбһңбһ·бһ‘бҹ’бһҷбһ¶**бҹ– бһўбһ¶бһӣбҹӢбһ бҹ’бһӮбһҡбһёбһ’бһҳбҹҚбһ…бҹҶбһҺбһ»бһ…бһҶбҹ’бһӣбһ„бһҠбһ»бҹҶбһ–бһҸбҹӢ
- **бһ”бһ„бҹ’бһҖбһҫбһ“бһ”бҹ’бһҡбһҹбһ·бһ‘бҹ’бһ’бһ—бһ¶бһ–**бҹ– бһҡбһ…бһ“бһ¶бһҹбһҳбҹ’бһ–бҹҗбһ“бҹ’бһ’бһ’бҹ’бһңбһҫбһҸбҹҒбһҹбҹ’бһҸбһ”бҹүбҹҮбһ‘бһ„бҹ’бһӮбһ·бһ…бһӣбһҝбһ“
- **бһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„бһҳбҹүбҹҒбһҳбҹүбһјбһҡбһё**бҹ– бһӣбҹҶбһ“бһ¶бҹҶбһҹбһ»бһңбһҸбҹ’бһҗбһ·бһ—бһ¶бһ–бһҖбҹ’бһ“бһ»бһ„бһҖбһ¶бһҡбһ”бһ„бҹ’бһҖбһҫбһҸбһ“бһ·бһ„бһ”бҹҶбһ•бҹ’бһӣбһ¶бһүбһңбһҸбҹ’бһҗбһ»
- **бһҖбһ¶бһҡбһҸбҹ’бһҡбһҪбһҸбһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһҜбһҖбһҸбҹ’бһҸбһҖбһҳбҹ’бһҳ**бҹ– бһҖбһ¶бһҡбһ‘бһ¶бһҖбҹӢбһ‘бһ„бһӮбҹ’бһҳбһ¶бһ“бһҖбһ¶бһҡбһ…бһ„бһҒбҹ’бһҹбҹӮбһ”бһ“бҹ’бһ‘бһ¶бһҸбҹӢ
