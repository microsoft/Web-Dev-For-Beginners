**бһҖбһјбһҠбһҹбһ¶бһҖбһӣбҹ’бһ”бһ„бһҒбҹ’бһӣбһё**:

бһўбҹ’бһ“бһҖбһҸбҹ’бһҡбҹҖбһҳбһҒбҹ’бһӣбһҪбһ“бһҡбһҪбһ…бһ бһҫбһҷбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһ”бһ„бҹ’бһҖбһҫбһҸбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһё AI - бһўбһҹбҹ’бһ…бһ¶бһҡбҹ’бһҷ! бһ”бҹүбһ»бһ“бҹ’бһҸбҹӮбһ“бҹҒбҹҮбһҮбһ¶бһҡбһҝбһ„бҹ– бһўбҹ’бһ“бһҖбһҳбһ¶бһ“бһ•бҹ’бһӣбһјбһңбһҮбһ¶бһ…бҹ’бһҡбһҫбһ“бһҠбҹӮбһӣбһўбһ¶бһ…бһҮбҹ’бһҡбһҫбһҹбһҡбһҫбһҹбһ”бһ¶бһ“ бһ бһҫбһҷбһҡбһ¶бһӣбҹӢбһ•бҹ’бһӣбһјбһңбһ‘бһ¶бҹҶбһ„бһ“бҹ„бҹҮбһҳбһ¶бһ“бһўбһҸбҹ’бһҗбһ”бҹ’бһҡбһҷбҹ„бһҮбһ“бҹҚбһ“бһ·бһ„бһӮбҹ’бһҡбҹ„бҹҮбһҗбҹ’бһ“бһ¶бһҖбҹӢбһ•бҹ’бһ‘бһ¶бһӣбҹӢбһҒбҹ’бһӣбһҪбһ“бҹ” бһңбһ¶бһҠбһјбһ…бһҮбһ¶бһҖбһ¶бһҡбһҮбҹ’бһҡбһҫбһҹбһҡбһҫбһҹбһҡбһңбһ¶бһ„бһҠбһҫбһҡбһҮбһҫбһ„ бһҹбҹҖбһҖбһҖбһ„бҹӢ бһ¬бһ”бһҫбһҖбһЎбһ¶бһ“бһ‘бҹ…бһҖбһ“бҹ’бһӣбҹӮбһ„бһҺбһ¶бһҳбһҪбһҷ - бһ‘бһ¶бҹҶбһ„бһўбһҹбҹӢбһҸбҹ’бһҡбһјбһңбһ“бһ¶бҹҶбһўбҹ’бһ“бһҖбһұбҹ’бһҷбһ‘бҹ…бһҠбһӣбҹӢ бһӮбһәбһҸбҹӮбһ”бһ‘бһ–бһ·бһҹбҹ„бһ’бһ“бҹҚ (бһ“бһ·бһ„бһҖбһҳбҹ’бһӣбһ¶бҹҶбһ„бһ бһ¶бһҸбҹӢбһ”бҹ’бһҡбһ¶бһҺ) бһ“бһ№бһ„бһҒбһ»бһҹбһӮбҹ’бһ“бһ¶бһ‘бҹ…бһҳбһҪбһҷбһҷбҹүбһ¶бһ„бһҳбһҪбһҷбҹ”

### бһ”бһ„бҹ’бһҖбһҫбһҸбһҖбһ¶бһҡбһҹбһ“бҹ’бһ‘бһ“бһ¶бһҡбһ”бһҹбҹӢбһўбҹ’бһ“бһҖбһҠбҹҶбһ”бһјбһ„

бһҸбҹ„бҹҮбһ”бһ„бҹ’бһҖбһҫбһҸбһҖбһ¶бһҡбһҹбһ“бҹ’бһ‘бһ“бһ¶бһҠбҹ„бһҷ AI бһҡбһ”бһҹбҹӢбһҷбһҫбһ„бһ…бһјбһӣбһҡбһҪбһҳбһҮбһ¶бһҸбһҪбһ”бһ»бһӮбҹ’бһӮбһӣбһҮбһ¶бһҖбҹӢбһӣбһ¶бһҖбҹӢбҹ” бһҷбһҫбһ„бһ“бһ№бһ„бһўбҹ„бһҷбһңбһ¶бһ”бҹ’бһҡбһҹбһ·бһ‘бҹ’бһ’бһ—бһ¶бһ–бһҮбһ¶бһ‘бһ¶бһ бһ¶бһ“бһҲбҹ’бһҳбҹ„бҹҮ Captain Picard - бһҸбһҪбһўбһ„бҹ’бһӮбһҠбҹӮбһӣбһҳбһ¶бһ“бһ–бһҡбһҮбһ¶бһҡбһ”бһҹбҹӢбһ…бҹҶбһҺбҹҒбҹҮбһҠбһ№бһ„бһ“бһҷбҹ„бһ”бһ¶бһҷ бһ“бһ·бһ„бһ—бһ¶бһ–бһҠбһ№бһҖбһ“бһ¶бҹҶбҹ–

**бһ”бҹҶбһ”бҹӮбһҖбһҖбһ¶бһҡбһҡбҹҖбһ”бһ…бҹҶбһҖбһ¶бһҡбһҹбһ“бҹ’бһ‘бһ“бһ¶бһ“бҹҒбҹҮбҹ–**
- **бһҖбҹҶбһҺбһҸбҹӢ** бһҸбһҪбһҡбһ“бһ·бһ„бһўбһҸбҹ’бһҸбһҹбһүбҹ’бһүбһ¶бһҺ AI бһҸбһ¶бһҳ `SystemMessage`
- **бһ•бҹ’бһҸбһӣбҹӢ** бһҹбҹҶбһҺбһҪбһҡбһҠбҹҶбһ”бһјбһ„бһўбҹ’бһ“бһҖбһ”бҹ’бһҡбһҫбһҸбһ¶бһҳ `HumanMessage`
- **бһ”бһ„бҹ’бһҖбһҫбһҸ** бһҳбһјбһӣбһҠбҹ’бһӢбһ¶бһ“бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҖбһ¶бһҡбһҹбһ“бҹ’бһ‘бһ“бһ¶бһ…бҹ’бһҡбһҫбһ“бһҮбҹҗбһҷ

бһҖбһјбһҠбһ–бҹҒбһүбһӣбҹҒбһүбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһ§бһ‘бһ¶бһ бһҡбһҺбҹҚбһ“бҹҒбҹҮбһ бһҫбһҷбһҳбһ¶бһ“бһҡбһјбһ”бһҡбһ¶бһ„бһҠбһјбһ…бһҮбһ¶бҹ–

```python
from langchain_core.messages import HumanMessage, SystemMessage
from langchain_openai import ChatOpenAI
import os

llm = ChatOpenAI(
    api_key=os.environ["GITHUB_TOKEN"],
    base_url="https://models.github.ai/inference",
    model="openai/gpt-4o-mini",
)

messages = [
    SystemMessage(content="You are Captain Picard of the Starship Enterprise"),
    HumanMessage(content="Tell me about you"),
]


# бһ’бҹ’бһңбһҫбһҖбһ¶бһҡ
response  = llm.invoke(messages)
print(response.content)
```

бһ’бҹ’бһӣбһ¶бһ”бҹӢбһӮбһ·бһҸбһ‘бҹҒбһҗбһ¶ ChatGPT бһҳбһҫбһӣбһ‘бҹ… "бһңбһ¶бһҷбһўбһҖбҹ’бһҹбһҡбһҡбһ”бһҹбҹӢбһңбһ¶" бһҮбһ¶бһҖбһ¶бһҡбһ–бһ·бһҸбһ”бһ“бҹ’бһҸбһ•бҹ’бһ‘бһ¶бһӣбҹӢбһҡбһ бһјбһҸбһҠбһӣбҹӢбһ…бһ”бҹӢ? бһ“бҹҒбҹҮбһӮбһәбһҹбҹ’бһ‘бҹ’бһҡбһёбһҳбһҖбҹ’бһ“бһ»бһ„бһҹбһҖбһҳбҹ’бһҳбһ—бһ¶бһ–бҹ” бһҠбһјбһ…бһҮбһ¶бһҖбһ¶бһҡбһҳбһҫбһӣбһўбҹ’бһ“бһҖбһҹбһҡбһҹбҹҒбһҡбһҗбҹ’бһ“бһ¶бһҖбҹӢбһҮбҹҶбһ“бһ¶бһү - бһҳбһҫбһӣбһҸбһҪбһўбһҖбҹ’бһҹбһҡбһ”бһ„бҹ’бһ бһ¶бһүбһҳбҹ’бһҸбһ„бһҳбҹ’бһҠбһ„ бһҮбһ¶бһҮбҹҶбһ бһ¶бһ“бһҠбһјбһ…бһҮбһ¶бһ…бһ»бһ“ бһҳбһ·бһ“бһҳбҹӮбһ“бһҖбһҫбһҸбһЎбһҫбһ„бһ—бҹ’бһӣбһ¶бһҳбҹ— - бһҹбҹ’бһ‘бҹ’бһҡбһёбһҳбһ’бҹ’бһңбһҫбһІбҹ’бһҷбһўбһ¶бһҡбһҳбҹ’бһҳбһҺбҹҚбһ–бһ¶бһҖбҹӢбһ–бҹҗбһ“бҹ’бһ’бһ’бһҳбҹ’бһҳбһҮбһ¶бһҸбһ· бһҳбһ·бһ“бһІбҹ’бһҷбһҳбһ¶бһ“бһҖбһ¶бһҡбһҡбһ„бҹӢбһ…бһ¶бҹҶбһҷбһјбһҡбһ“бһ·бһ„бһ•бҹ’бһҸбһӣбҹӢбһҳбҹ’бһ…бһ¶бһҹбҹӢбһ”бҹ’бһҡбһҫбһ”бҹ’бһҡбһ¶бһҹбҹӢбһІбҹ’бһҷбһҶбһ¶бһ”бҹӢбһҶбҹ’бһӣбһҫбһҷбһҸбһ”бҹ”

### бһўбһ“бһ»бһңбһҸбҹ’бһҸбһҹбҹ’бһ‘бҹ’бһҡбһёбһҳбһҮбһ¶бһҳбһҪбһҷ LangChain

```python
from langchain_openai import ChatOpenAI
import os

llm = ChatOpenAI(
    api_key=os.environ["GITHUB_TOKEN"],
    base_url="https://models.github.ai/inference",
    model="openai/gpt-4o-mini",
    streaming=True
)

# бһҹбҹ’бһ‘бҹ’бһҡбһёбһҳбһ…бһҳбҹ’бһӣбһҫбһҷ
for chunk in llm.stream("Write a short story about a robot learning to code"):
    print(chunk.content, end="", flush=True)
```

**бһ бҹҒбһҸбһ»бһўбҹ’бһңбһёбһ”бһ¶бһ“бһҮбһ¶бһҹбҹ’бһ‘бҹ’бһҡбһёбһҳбһўбһҹбҹ’бһ…бһ¶бһҡбҹ’бһҷбҹ–**
- **бһ”бһ„бҹ’бһ бһ¶бһү** бһҳбһ¶бһҸбһ·бһҖбһ¶бһ“бҹ…бһ–бҹҒбһӣбһңбһ¶бһҖбҹҶбһ–бһ»бһ„бһ”бһ„бҹ’бһҖбһҫбһҸ - бһҳбһ·бһ“бһ…бһ¶бҹҶбһ”бһ¶бһ…бҹӢбһҡбһ„бҹӢбһ…бһ¶бҹҶбһҷбһјбһҡ!
- **бһ’бҹ’бһңбһҫбһІбҹ’бһҷ** бһўбҹ’бһ“бһҖбһ”бҹ’бһҡбһҫбһҳбһ¶бһ“бһўбһ¶бһҡбһҳбҹ’бһҳбһҺбҹҚбһҗбһ¶бһҳбһ¶бһ“бһўбҹ’бһңбһёбһҖбҹҶбһ–бһ»бһ„бһҖбһҫбһҸбһЎбһҫбһ„бһ–бһ·бһҸ
- **бһҳбһ¶бһ“бһўбһ¶бһҡбһҳбҹ’бһҳбһҺбҹҚ** бһӣбһҝбһ“ бһ‘бҹ„бҹҮбһ”бһёбһңбһ¶бһҳбһ·бһ“бһӣбһҝбһ“бһ–бһ·бһҸбһҖбҹҸбһҠбҹ„бһҷ
- **бһўбһ“бһ»бһүбҹ’бһүбһ¶бһҸ** бһўбҹ’бһ“бһҖбһ”бҹ’бһҡбһҫбһ…бһ¶бһ”бҹӢбһ•бҹ’бһҸбһҫбһҳбһўбһ¶бһ“ бһҒбһҺбҹҲбһҠбҹӮбһӣ AI бһҖбҹҶбһ–бһ»бһ„ "бһӮбһ·бһҸ"

> рҹ’Ў **бһҮбҹҶбһ“бһҪбһҷбһ”бһ‘бһ–бһ·бһҹбҹ„бһ’бһ“бҹҚбһўбҹ’бһ“бһҖбһ”бҹ’бһҡбһҫбҹ–** бһҹбҹ’бһ‘бҹ’бһҡбһёбһҳбһ–бһ·бһҸбһҮбһ¶бһ”бҹ’бһҡбһҹбһҫбһҡбһҠбҹ„бһҷбһҹбһ¶бһҡбһўбҹ’бһ“бһҖбһ”бһ¶бһ“бһ…бҹҶбһҺбһ¶бһҷбһ–бҹҒбһӣбһҮбһ¶бһҳбһҪбһҷбһ…бһҳбҹ’бһӣбһҫбһҷбһңбҹӮбһ„бҹ—бһҠбһјбһ…бһҮбһ¶ бһҖбһ¶бһҡбһ–бһ“бҹ’бһҷбһӣбҹӢбһҖбһјбһҠ бһҖбһ¶бһҡбһҹбһҡбһҹбҹҒбһҡбһ…бҹ’бһ“бҹғбһ”бҹ’бһҡбһҢбһ·бһҸ бһ¬бһҳбҹҒбһҡбҹҖбһ“бһӣбһҳбҹ’бһўбһ·бһҸбҹ” бһўбҹ’бһ“бһҖбһ”бҹ’бһҡбһҫбһ”бҹ’бһҡбһ¶бһҹбҹӢбһҡбһ”бһҹбҹӢбһўбҹ’бһ“бһҖбһ“бһ№бһ„бһҹбҹ’бһҡбһЎбһ¶бһүбҹӢбһҳбһҫбһӣбһҖбһ¶бһҡбһңбһ·бһңбһҸбҹ’бһҸбһ“бҹҚбһҮбҹҶбһ“бһҪбһҹбһҖбһ¶бһҡбһҳбһҫбһӣбһўбҹҒбһҖбҹ’бһҡбһ„бҹӢбһ’бҹ’бһңбһҫбһҡбһңбһ¶бһҖбҹӢбһҡбһңбһӣбҹӢ!

### рҹҺҜ бһҖбһ¶бһҡбһҸбҹ’бһҡбһҪбһҸбһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһ•бҹ’бһ“бҹӮбһҖбһўбһ”бҹӢбһҡбҹҶбҹ– бһўбһҸбҹ’бһҗбһ”бҹ’бһҡбһҷбҹ„бһҮбһ“бҹҚбһҡбһ”бһҹбҹӢбһҖбһҳбҹ’бһҡбһ·бһҸбһҹбһ“бҹ’бһҠбһ·бһҹбһ»бһҒбһҹбҹҠбһ»бһҳбһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’

**бһҲбһ”бҹӢ бһ“бһ·бһ„бһ–бһ·бһ…бһ¶бһҡбһҺбһ¶**бҹ– бһўбҹ’бһ“бһҖбһ‘бһҫбһ”бһ”бҹүбҹҮбһ–бһ¶бһӣбҹӢбһҗбһ¶бһҳбһ–бһӣбһҡбһ”бһҹбҹӢбһҖбһҳбҹ’бһҡбһ·бһҸбһҹбһ“бҹ’бһҠбһ·бһҹбһ»бһҒбһҹбҹҠбһ»бһҳбһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’ AIбҹ” бһ”бҹ’бһҡбһҖбһҪбһҸбһ’бҹҖбһ”бһ‘бҹ…бһ“бһ№бһ„бһҖбһ¶бһҡбһ бҹ… API бһ•бҹ’бһ‘бһ¶бһӣбҹӢбһ–бһёбһҳбҹҒбһҡбҹҖбһ“бһҳбһ»бһ“бҹ”

**бһҖбһ¶бһҡбһңбһ¶бһҷбһҸбһҳбҹ’бһӣбҹғбһҒбҹ’бһӣбһҪбһ“бһҜбһ„бһҷбҹүбһ¶бһ„бһҶбһ¶бһ”бҹӢбһҡбһ бҹҗбһҹбҹ–**
- бһҸбһҫбһўбҹ’бһ“бһҖбһўбһ¶бһ…бһ–бһҺбҹҢбһ“бһ¶бһҗбһ¶ LangChain бһ”бһүбҹ’бһ…бҹҒбһүбһ—бһ¶бһ–бһ„бһ¶бһҷбһҹбҹ’бһҡбһҪбһӣбһҖбҹ’бһ“бһ»бһ„бһҖбһ¶бһҡбһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„бһҹбһ“бҹ’бһ‘бһ“бһ¶бһҷбҹүбһ¶бһ„бһҠбһјбһ…бһҳбҹ’бһҸбҹҒбһ… бһ”бҹ’бһҡбҹҖбһ”бһ’бҹҖбһ”бһ“бһ№бһ„бһҖбһ¶бһҡбһҸбһ¶бһҳбһҠбһ¶бһ“бһҹбһ¶бһҡбһҠбҹ„бһҷбһҠбҹғ?

### бһҖбҹҶбһҺбһҸбҹӢ schema бһӣбһ‘бҹ’бһ’бһ•бһӣ

```python
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import JsonOutputParser
from pydantic import BaseModel, Field

class CodeReview(BaseModel):
    score: int = Field(description="Code quality score from 1-10")
    strengths: list[str] = Field(description="List of code strengths")
    improvements: list[str] = Field(description="List of suggested improvements")
    overall_feedback: str = Field(description="Summary feedback")

# бһҖбҹҶбһҺбһҸбҹӢбһўбҹ’бһ“бһҖбһңбһ·бһ—бһ¶бһӮ
parser = JsonOutputParser(pydantic_object=CodeReview)

# бһ”бһ„бҹ’бһҖбһҫбһҸбһҖбһ¶бһҡбһҮбҹҶбһҡбһ»бһүбһҮбһ¶бһҳбһҪбһҷбһ“бһ№бһ„бһҖбһ¶бһҡбһҺбҹӮбһ“бһ¶бҹҶбһ‘бҹҶбһҡбһ„бҹӢ
prompt = ChatPromptTemplate.from_messages([
    ("system", "You are a code reviewer. {format_instructions}"),
    ("human", "Review this code: {code}")
])

# бһҠбһ¶бһҖбҹӢбһ‘бҹҶбһҡбһ„бҹӢбһҖбһ¶бһҡбһҮбҹҶбһҡбһ»бһүбһҮбһ¶бһҳбһҪбһҷбһ“бһ№бһ„бһҖбһ¶бһҡбһҺбҹӮбһ“бһ¶бҹҶ
chain = prompt | llm | parser

# бһ‘бһ‘бһҪбһӣбһ”бһ¶бһ“бһ…бһҳбҹ’бһӣбһҫбһҷбһҠбҹӮбһӣбһҳбһ¶бһ“бһҡбһ…бһ“бһ¶бһҹбһҳбҹ’бһ–бҹҗбһ“бҹ’бһ’
code_sample = """
def calculate_average(numbers):
    return sum(numbers) / len(numbers)
"""

result = chain.invoke({
    "code": code_sample,
    "format_instructions": parser.get_format_instructions()
})

print(f"Score: {result['score']}")
print(f"Strengths: {', '.join(result['strengths'])}")
```

**бһ бҹҒбһҸбһ»бһўбҹ’бһңбһёбһ”бһ¶бһ“бһҮбһ¶бһӣбһ‘бҹ’бһ’бһ•бһӣбһҡбҹҖбһ”бһ…бҹҶбһҮбһ¶бһҮбһҳбҹ’бһҡбһҫбһҹбһӮбһ¶бҹҶбһ‘бҹ’бһҡбҹ–**
- **бһ”бһүбҹ’бһ…бһ”бҹӢ** бһҖбһ¶бһҡбһ”бҹүбһ¶бһ“бҹӢбһҹбҹ’бһҳбһ¶бһ“бһҗбһ¶бһҸбһҫбһўбҹ’бһ“бһҖбһ“бһ№бһ„бһ‘бһ‘бһҪбһӣбһ”бһ¶бһ“бһ‘бҹ’бһҡбһ„бҹӢбһ‘бҹ’бһҡбһ¶бһҷбһўбҹ’бһңбһё - бһңбһ¶бһӮбҹ’бһҡбһ”бҹӢбһ–бҹҒбһӣбһҸбҹ’бһҡбһ№бһҳбһҸбҹ’бһҡбһјбһңбһҠбҹӮбһҡ
- **бһ—бҹ’бһҮбһ¶бһ”бҹӢ** бһҸбҹ’бһҡбһ„бҹӢбһ‘бҹ…бһҳбһјбһӣбһҠбҹ’бһӢбһ¶бһ“бһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷ бһ“бһ·бһ„ API бһҷбҹүбһ¶бһ„бһ„бһ¶бһҷ бһҹбҹ’бһҡбһҪбһӣ
- **бһ…бһ¶бһ”бҹӢбһҷбһҖ** бһ…бһҳбҹ’бһӣбһҫбһҷбһ’бһ»бһүбһҗбһ”бҹӢбһҡбһ”бһҹбҹӢ AI бһҳбһ»бһ“бһ–бҹҒбһӣбһңбһ¶бһ”бҹҶбһ•бҹ’бһӣбһ¶бһүбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһўбҹ’бһ“бһҖ

### бһ”бҹ’бһҡбһҫбһ”бҹ’бһҡбһ¶бһҹбҹӢ Python

бһҸбҹ„бҹҮбһ”бһ“бҹ’бһҗбҹӮбһҳбһ§бһ”бһҖбһҡбһҺбҹҚбһҳбһҪбһҷбһ…бҹҶбһ“бһҪбһ“бһҠбһјбһ…бһҒбһ¶бһ„бһҖбҹ’бһҡбҹ„бһҳбҹ–

```python
from typing_extensions import Annotated, TypedDict

class add(TypedDict):
    """Add two integers."""

    # бһҖбһ¶бһҡбһҳбҹүбһ¶бһҖбҹӢбһҹбҹ’бһӣбһ¶бһҖбһҸбҹ’бһҡбһјбһңбһҸбҹӮбһҳбһ¶бһ“бһ”бҹ’бһҡбһ—бҹҒбһ‘ бһ бһҫбһҷбһўбһ¶бһ…бһҡбһҪбһҳбһ”бһүбҹ’бһ…бһјбһӣбһҸбһҳбҹ’бһӣбҹғбһӣбҹҶбһ“бһ¶бҹҶбһҠбһҫбһҳ бһ“бһ·бһ„бһҖбһ¶бһҡбһ–бһҺбҹҢбһ“бһ¶бһ”бҹ’бһҡбһҹбһ·бһ“бһ”бһҫбһ…бһ¶бһ”бҹӢбһўбһ¶бһҡбһҳбҹ’бһҳбһҺбҹҚ (бһҸбһ¶бһҳбһӣбҹҶбһҠбһ¶бһ”бҹӢбһ“бҹ„бҹҮ)бҹ”
    a: Annotated[int, ..., "First integer"]
    b: Annotated[int, ..., "Second integer"]

tools = [add]

functions = {
    "add": lambda a, b: a + b
}
```

бһҸбһҫбһҳбһ¶бһ“бһўбҹ’бһңбһёбһҖбһҫбһҸбһЎбһҫбһ„бһ‘бһёбһ“бҹҒбҹҮ? бһҷбһҫбһ„бһҖбҹҶбһ–бһ»бһ„бһ”бһ„бҹ’бһҖбһҫбһҸбһӮбҹҶбһ“бһјбһҹбһҡбһјбһ”бһҳбһ“бҹ’бһҸбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһ§бһ”бһҖбһҡбһҺбҹҚбһҳбһҪбһҷбһ бҹ…бһҗбһ¶ `add`бҹ” бһҠбҹ„бһҷбһ‘бһ‘бһҪбһӣбһҳбһҖбһ–бһё `TypedDict` бһ“бһ·бһ„бһ”бҹ’бһҡбһҫбһ”бҹ’бһҡбһ—бҹҒбһ‘ `Annotated` бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢ `a` бһ“бһ·бһ„ `b` бһ—бһ¶бһҹбһ¶бһҠбһјбһ…бһ“бҹҒбҹҮбһ•бҹ’бһҠбһӣбҹӢбһҡбһјбһ”бһҳбһ“бҹ’бһҸбһ…бҹ’бһ”бһ¶бһҹбҹӢбһӣбһ¶бһҹбҹӢбһ‘бҹ… LLM бһўбҹҶбһ–бһёбһўбҹ’бһңбһёбһҠбҹӮбһӣбһ§бһ”бһҖбһҡбһҺбҹҚбһ“бҹҒбҹҮбһ’бҹ’бһңбһҫ бһ“бһ·бһ„бһҸбһҫбһңбһ¶бһҸбҹ’бһҡбһјбһңбһҖбһ¶бһҡвҖӢбһўбҹ’бһңбһёбҹ” бһңбһ…бһ“бһ¶бһҖбһ¶бһҡбһңбһ·бһ’бһёбһҹбһ¶бһҹбҹ’бһҸбҹ’бһҡ `functions` бһҠбһјбһ…бһҮбһ¶бһ”бһҺбҹ’бһҠбһ»бҹҶбһ§бһ”бһҖбһҡбһҺбҹҚбһҡбһ”бһҹбҹӢбһҷбһҫбһ„ - бһңбһ¶бһҗбҹ’бһӣбҹӮбһ„бһ–бһёбһҖбһ¶бһҡвҖӢбһўбһ“бһ»бһңбһҸбҹ’бһҸбһҖбһјбһҠбһ–бҹҒбһӣбһҠбҹӮбһӣ AI бһҹбһҳбҹ’бһҡбҹҒбһ…бһ…бһ·бһҸбҹ’бһҸбһ”бҹ’бһҡбһҫбһ§бһ”бһҖбһҡбһҺбҹҚбһҮбһ¶бһҖбҹӢбһӣбһ¶бһҖбҹӢбһҳбһҪбһҷбҹ”

бһҸбҹ„бҹҮбһҳбһҫбһӣбһҡбһ”бҹҖбһ”бһҠбҹӮбһӣбһҷбһҫбһ„бһ бҹ… LLM бһҮбһ¶бһҳбһҪбһҷбһ§бһ”бһҖбһҡбһҺбҹҚбһ“бҹҒбҹҮбһ”бһ“бҹ’бһ‘бһ¶бһ”бҹӢбҹ–

```python
llm = ChatOpenAI(
    api_key=os.environ["GITHUB_TOKEN"],
    base_url="https://models.github.ai/inference",
    model="openai/gpt-4o-mini",
)

llm_with_tools = llm.bind_tools(tools)
```

бһ“бҹ…бһ‘бһёбһ“бҹҒбҹҮ бһҷбһҫбһ„бһ бҹ… `bind_tools` бһҮбһ¶бһҳбһҪбһҷбһўбһ¶бһҡбҹүбҹҒ `tools` бһҡбһ”бһҹбҹӢбһҷбһҫбһ„ бһ бһҫбһҷ LLM `llm_with_tools` бһ“бҹҒбҹҮбһҳбһ¶бһ“бһ…бҹҶбһҺбҹҒбҹҮбһҠбһ№бһ„бһўбҹҶбһ–бһёбһ§бһ”бһҖбһҡбһҺбҹҚбһ“бҹҒбҹҮбһҡбһҪбһ…бһ бһҫбһҷбҹ”

бһҠбһҫбһҳбҹ’бһ”бһёбһ”бҹ’бһҡбһҫбһ”бҹ’бһҡбһ¶бһҹбҹӢ LLM бһҗбҹ’бһҳбһёбһ“бҹҒбҹҮ бһҷбһҫбһ„бһўбһ¶бһ…бһңбһ¶бһҷбһҖбһјбһҠбһҠбһјбһ…бһҒбһ¶бһ„бһҖбҹ’бһҡбҹ„бһҳбҹ–

```python
query = "What is 3 + 12?"

res = llm_with_tools.invoke(query)
if(res.tool_calls):
    for tool in res.tool_calls:
        print("TOOL CALL: ", functions[tool["name"]](../../../10-ai-framework-project/**tool["args"]))
print("CONTENT: ",res.content)
```

бһҘбһЎбһјбһңбһ“бҹҒбҹҮбһ–бҹҒбһӣбһҷбһҫбһ„бһ бҹ… `invoke` бһӣбһҫ llm бһҗбҹ’бһҳбһёбһҠбҹӮбһӣбһҳбһ¶бһ“бһ§бһ”бһҖбһҡбһҺбҹҚбһ“бҹҒбҹҮ бһ бһҫбһҷбһ”бҹ’бһҡбһҹбһ·бһ“бһ”бһҫбһҳбһ¶бһ“бһӮбһ»бһҺбһӣбһҖбҹ’бһҒбһҺбҹҲ `tool_calls` бһ”бһ¶бһ“бһ”бҹҶбһ–бҹҒбһүбҹ” бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһ§бһ”бһҖбһҡбһҺбҹҚбһҺбһ¶бһҳбһҪбһҷбһҠбҹӮбһӣбһ”бһ¶бһ“бһҖбҹҶбһҺбһҸбҹӢ бһҳбһ¶бһ“бһӣбһҖбҹ’бһҒбһҺбҹҲ `name` бһ“бһ·бһ„ `args` бһ”бһүбҹ’бһҮбһ¶бһҖбҹӢбһҗбһ¶бһ§бһ”бһҖбһҡбһҺбҹҚбһҺбһ¶бһӮбһҪбһҡбһҸбҹ’бһҡбһјбһңбһ”бһҺбҹ’бһҠбһҫбһҡбһ бҹ… бһ“бһ·бһ„бһҮбһ¶бһҳбһҪбһҷбһўбһ¶бһӮбһ»бһҷбһҳбҹүбһ„бҹӢбҹ” бһҖбһјбһҠбһ–бҹҒбһүбһӣбҹҒбһүбһҳбһ¶бһ“бһҡбһ¶бһ„бһҠбһјбһ…бһ“бҹҒбҹҮбҹ–

### бһҖбһ¶бһҡбһ”бһ„бҹ’бһҖбһҫбһҸ бһ“бһ·бһ„бһ”бҹ’бһҡбһҫ embeddings

```python
from langchain_openai import OpenAIEmbeddings
from langchain_community.vectorstores import FAISS
from langchain_community.document_loaders import TextLoader
from langchain.text_splitter import CharacterTextSplitter

# бһ…бһ¶бһ”бҹӢбһ•бҹ’бһҸбһҫбһҳбһ”бһ„бҹ’бһҖбһҫбһҸ embeddings
embeddings = OpenAIEmbeddings(
    api_key=os.environ["GITHUB_TOKEN"],
    base_url="https://models.github.ai/inference",
    model="text-embedding-3-small"
)

# бһ•бҹ’бһ‘бһ»бһҖ бһ“бһ·бһ„ бһ”бҹҶбһ”бҹӮбһҖбһҜбһҖбһҹбһ¶бһҡ
loader = TextLoader("documentation.txt")
documents = loader.load()

text_splitter = CharacterTextSplitter(chunk_size=1000, chunk_overlap=0)
texts = text_splitter.split_documents(documents)

# бһ”бһ„бҹ’бһҖбһҫбһҸбһ бһ¶бһ„бһңбҹүбһ·бһ…бһ‘бҹҗбһҡ
vectorstore = FAISS.from_documents(texts, embeddings)

# бһ’бҹ’бһңбһҫбһҖбһ¶бһҡбһҹбҹ’бһңбҹӮбһ„бһҡбһҖбһҠбҹ„бһҷбһ•бҹ’бһўбҹӮбһҖбһӣбһҫбһ—бһ¶бһ–бһҹбҹ’бһҡбһҠбҹҖбһ„
query = "How do I handle user authentication?"
similar_docs = vectorstore.similarity_search(query, k=3)

for doc in similar_docs:
    print(f"Relevant content: {doc.page_content[:200]}...")
```

### бһ§бһ‘бһ¶бһ бһҡбһҺбҹҚбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһ–бҹҒбһүбһӣбҹҒбһү

```python
from langchain_openai import ChatOpenAI, OpenAIEmbeddings
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.messages import HumanMessage, SystemMessage, AIMessage
from langchain_community.vectorstores import FAISS
from typing_extensions import Annotated, TypedDict
import os
import requests

class CodingAssistant:
    def __init__(self):
        self.llm = ChatOpenAI(
            api_key=os.environ["GITHUB_TOKEN"],
            base_url="https://models.github.ai/inference",
            model="openai/gpt-4o-mini"
        )
        
        self.conversation_history = [
            SystemMessage(content="""You are an expert coding assistant. 
            Help users learn programming concepts, debug code, and write better software.
            Use tools when needed and maintain a helpful, encouraging tone.""")
        ]
        
        # бһҖбҹҶбһҺбһҸбҹӢбһ§бһ”бһҖбһҡбһҺбҹҚ
        self.setup_tools()
    
    def setup_tools(self):
        class web_search(TypedDict):
            """Search for programming documentation or examples."""
            query: Annotated[str, "Search query for programming help"]
        
        class code_formatter(TypedDict):
            """Format and validate code snippets."""
            code: Annotated[str, "Code to format"]
            language: Annotated[str, "Programming language"]
        
        self.tools = [web_search, code_formatter]
        self.llm_with_tools = self.llm.bind_tools(self.tools)
    
    def chat(self, user_input: str):
        # бһ”бһ“бҹ’бһҗбҹӮбһҳбһҹбһ¶бһҡбһҡбһ”бһҹбҹӢбһўбҹ’бһ“бһҖбһ”бҹ’бһҡбһҫбһ‘бҹ…бһҖбһ¶бһ“бҹӢбһҖбһ¶бһҡбһҹбһ“бҹ’бһ‘бһ“бһ¶
        self.conversation_history.append(HumanMessage(content=user_input))
        
        # бһ‘бһ‘бһҪбһӣбһ”бһ¶бһ“бһҖбһ¶бһҡбһҶбҹ’бһӣбһҫбһҷбһҸбһ”бһ–бһё AI
        response = self.llm_with_tools.invoke(self.conversation_history)
        
        # бһҠбҹ„бҹҮбһҹбҹ’бһҡбһ¶бһҷбһҖбһ¶бһҡбһ бҹ…бһ§бһ”бһҖбһҡбһҺбҹҚ бһ”бҹ’бһҡбһҹбһ·бһ“бһ”бһҫбһҳбһ¶бһ“
        if response.tool_calls:
            for tool_call in response.tool_calls:
                tool_result = self.execute_tool(tool_call)
                print(f"рҹ”§ Tool used: {tool_call['name']}")
                print(f"рҹ“Ҡ Result: {tool_result}")
        
        # бһ”бһ“бҹ’бһҗбҹӮбһҳбһҖбһ¶бһҡбһҶбҹ’бһӣбһҫбһҷбһҸбһ”бһ–бһё AI бһ‘бҹ…бһҖбһ¶бһ“бҹӢбһҖбһ¶бһҡбһҹбһ“бҹ’бһ‘бһ“бһ¶
        self.conversation_history.append(response)
        
        return response.content
    
    def execute_tool(self, tool_call):
        tool_name = tool_call['name']
        args = tool_call['args']
        
        if tool_name == 'web_search':
            return f"Found documentation for: {args['query']}"
        elif tool_name == 'code_formatter':
            return f"Formatted {args['language']} code: {args['code'][:50]}..."
        
        return "Tool execution completed"

# бһ§бһ‘бһ¶бһ бһҡбһҺбҹҚбһҖбһ¶бһҡбһ”бҹ’бһҡбһҫбһ”бһ¶бҹ’бһҡбһҹбҹӢ
assistant = CodingAssistant()

print("рҹӨ– Coding Assistant Ready! Type 'quit' to exit.\n")

while True:
    user_input = input("You: ")
    if user_input.lower() == 'quit':
        break
    
    response = assistant.chat(user_input)
    print(f"рҹӨ– Assistant: {response}\n")
```

**бһӮбҹҶбһ“бһ·бһҸбһҹбҹҶбһҒбһ¶бһ“бҹӢбҹ—бһҠбҹӮбһӣбһ”бһ¶бһ“бһ‘бһ‘бһҪбһӣбҹ–**
- **бһҡбһ…бһ“бһ¶бһҹбһҳбҹ’бһ–бҹҗбһ“бҹ’бһ’бһ•бҹ’бһўбҹӮбһҖбһӣбһҫбһҗбҹ’бһ“бһ¶бһҖбҹӢ**бҹ– бһҡбһ…бһ“бһ¶бһҹбһҳбҹ’бһ–бҹҗбһ“бҹ’бһ’бһҖбһҳбҹ’бһҳбһңбһ·бһ’бһё AI бһҠбҹӮбһӣбһҡбҹҖбһ”бһ…бҹҶбһ”бһ¶бһ“бһӣбҹ’бһў бһ“бһ·бһ„бһ„бһ¶бһҷбһҹбҹ’бһҡбһҪбһӣбһҗбҹӮбһҡбһҖбҹ’бһҹбһ¶
- **бһҖбһ¶бһҡбһҡбһҪбһҳбһ”бһүбҹ’бһ…бһјбһӣбһ§бһ”бһҖбһҡбһҺбҹҚ**бҹ– бһҳбһ»бһҒбһ„бһ¶бһҡбһ•бҹ’бһҹбҹҒбһ„бһ‘бҹҖбһҸбһҠбҹӮбһӣбһӣбһҫбһҹбһ–бһёбһҖбһ¶бһҡбһҹбһ“бҹ’бһ‘бһ“бһ¶
- **бһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„бһўбһ“бһ»бһҹбҹ’бһҹбһ¶бһңбһҡбһёбһҷбҹҚ**бҹ– бһ”бһҡбһ·бһ”бһ‘бһҹбһ“бҹ’бһ‘бһ“бһ¶бһ”бһ“бҹ’бһҸ
- **бһҖбһ¶бһҡбһҠбҹ„бҹҮбһҹбҹ’бһҡбһ¶бһҷбһҖбҹҶбһ бһ»бһҹ**бҹ– бһўбһ¶бһҖбһ”бҹ’бһ”бһҖбһ·бһҡбһ·бһҷбһ¶бһҖбһҳбҹ’бһҳбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһҠбҹҸбһҡбһ№бһ„бһҳбһ¶бҹҶ

**бһҖбһ¶бһҡбһҸбһ—бҹ’бһҮбһ¶бһ”бҹӢбһ§бһҹбҹ’бһҡбһ»бһҳбһҮбҹҶбһ“бһ¶бһү**бҹ– бһӣбҹҶбһ“бһ¶бҹҶбһҡбһ…бһ“бһ¶бһҹбһҳбҹ’бһ–бҹҗбһ“бҹ’бһ’бһҠбҹӮбһӣбһўбҹ’бһ“бһҖбһўбһ“бһ»бһңбһҸбҹ’бһҸ (бһҗбҹ’бһ“бһ¶бһҖбҹӢбһҹбһ“бҹ’бһ‘бһ“бһ¶ бһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’бһ§бһ”бһҖбһҡбһҺбҹҚ бһҖбһ¶бһҡбһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„бһўбһ“бһ»бһҹбҹ’бһҹбһ¶бһңбһҡбһёбһҷбҹҚ) бһӮбһәбһҮбһ¶бһӣбҹҶбһ“бһ¶бҹҶбһҠбһјбһ…бһӮбҹ’бһ“бһ¶бһ“бһ№бһ„бһҖбһҳбҹ’бһҳбһңбһ·бһ’бһё AI бһўбһ¶бһҮбһёбһңбһҖбһҳбҹ’бһҳбһҠбһјбһ…бһҮбһ¶ бһҮбҹҶбһ“бһҪбһҷбһҖбһ¶бһҡбһҡбһ”бһҹбҹӢ Slack, GitHub Copilot бһ“бһ·бһ„ Microsoft Copilotбҹ” бһўбҹ’бһ“бһҖбһҖбҹҶбһ–бһ»бһ„бһ”бһ„бҹ’бһҖбһҫбһҸбһҠбҹ„бһҷбһӮбһ·бһҸбһӮбһјбһҡбһҡбһ…бһ“бһ¶бһҹбһҳбҹ’бһ–бҹҗбһ“бҹ’бһ’бһҗбҹ’бһ“бһ¶бһҖбҹӢбһңбһ·бһҮбҹ’бһҮбһ¶бһҮбһёбһңбҹҲбҹ”

**бһҹбҹҶбһ“бһҪбһҡбһҸбҹ’бһҡбһӣбһ”бҹӢ**бҹ– бһҸбһҫбһўбҹ’бһ“бһҖбһ“бһ№бһ„бһ–бһ„бҹ’бһҡбһёбһҖбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһ“бҹҒбҹҮбһҠбһјбһ…бһҳбҹ’бһҸбҹҒбһ… бһҠбһҫбһҳбҹ’бһ”бһёбһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„бһўбҹ’бһ“бһҖбһ”бҹ’бһҡбһҫбһ”бҹ’бһҡбһ¶бһҹбҹӢбһ…бҹ’бһҡбһҫбһ“ бһ•бҹ’бһ‘бһ»бһҖбһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷбһ”бһ“бҹ’бһҸбһҮбһ¶бһ”бҹ’бһҡбһ…бһ¶бҹҶ бһ¬ бһҡбһҪбһҳбһ”бһүбҹ’бһ…бһјбһӣбһҮбһ¶бһҳбһҪбһҷбһҳбһјбһӣбһҠбҹ’бһӢбһ¶бһ“бһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷбһҒбһ¶бһ„бһҖбҹ’бһҡбҹ…? бһҹбһјбһҳбһӮбһ·бһҸбһ–бһёбһ”бһүбҹ’бһ бһ¶бһҖбһ¶бһҡбһ”бһ„бҹ’бһ бһ¶бһүбһ”бһҡбһ·бһҳбһ¶бһҺ бһ“бһ·бһ„бһҖбһ¶бһҡбһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„бһҡбһҠбҹ’бһӢбҹ”

## бһ—бһ¶бһҡбһҖбһ·бһ…бҹ’бһ…бҹ– бһ”бһ„бҹ’бһҖбһҫбһҸбһҮбҹҶбһ“бһҪбһҷбһҖбһ¶бһҡбһҹбһ·бһҖбҹ’бһҹбһ¶бһҮбһ¶бһҳбһҪбһҷ AI бһҡбһ”бһҹбҹӢбһўбҹ’бһ“бһҖбһҜбһ„

**бһӮбҹ„бһӣбһ”бҹҶбһҺбһ„**бҹ– бһ”бһ„бҹ’бһҖбһҫбһҸбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһё AI бһҮбһҪбһҷбһҹбһ·бһҹбҹ’бһҹбһҡбҹҖбһ“бһҳбҹҒбһҡбҹҖбһ“бһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһҠбҹ„бһҷбһ•бҹ’бһҸбһӣбҹӢбһҖбһ¶бһҡбһ–бһ“бҹ’бһҷбһӣбҹӢ бһ§бһ‘бһ¶бһ бһҡбһҺбҹҚбһҖбһјбһҠ бһ“бһ·бһ„бһҹбҹҶбһҺбһҪбһҡбһ”бҹ’бһҡбһЎбһ„бһ”бҹӮбһ”бһўбһ“бҹ’бһҸбһҡбһҖбһҳбҹ’бһҳбҹ”

### бһҸбһҳбҹ’бһҡбһјбһңбһҖбһ¶бһҡ

**бһӣбһҖбҹ’бһҒбһҺбҹҲбһҹбҹҶбһҒбһ¶бһ“бҹӢ (бһ…бһ¶бҹҶбһ”бһ¶бһ…бҹӢ):**
1. **бһ•бҹ’бһ‘бҹғбһҳбһ»бһҒбһҹбһ“бҹ’бһ‘бһ“бһ¶**бҹ– бһўбһ“бһ»бһңбһҸбҹ’бһҸбһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’бһҹбһ“бҹ’бһ‘бһ“бһ¶ бһҠбҹӮбһӣбһҡбһҖбҹ’бһҹбһ¶бһ”бһҡбһ·бһ”бһ‘бһҹбҹҶбһҺбһҪбһҡбһ–бһ бһ»бһҮбһҪбһҡ
2. **бһ§бһ”бһҖбһҡбһҺбҹҚбһўбһ”бҹӢбһҡбҹҶ**бҹ– бһ”бһ„бҹ’бһҖбһҫбһҸбһ§бһ”бһҖбһҡбһҺбҹҚбһ–бһёбһҡбһ…бҹҶбһ“бһҪбһ“бһҷбҹүбһ¶бһ„бһ бҹ„бһ…бһҺбһ¶бһҹбҹӢ бһҠбһҫбһҳбҹ’бһ”бһёбһҮбһҪбһҷбһҖбһ¶бһҡбһҹбһ·бһҖбҹ’бһҹбһ¶бҹ–
   - бһ§бһ”бһҖбһҡбһҺбҹҚбһ–бһ“бҹ’бһҷбһӣбҹӢбһҖбһјбһҠ
   - бһ§бһ”бһҖбһҡбһҺбҹҚбһ”бһ„бҹ’бһҖбһҫбһҸбһҹбҹҶбһҺбһҪбһҡбһ”бҹ’бһҡбһЎбһ„бһ–бһёбһҳбҹҒбһҡбҹҖбһ“
3. **бһҖбһ¶бһҡбһҹбһ·бһҖбҹ’бһҹбһ¶бһ”бһ»бһӮбҹ’бһӮбһӣбһ·бһҖ**бҹ– бһ”бҹ’бһҡбһҫбһҹбһ¶бһҡбһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’ бһҠбһҫбһҳбҹ’бһ”бһёбһўбһ“бһ»бһңбһҸбҹ’бһҸбһ…бһҳбҹ’бһӣбһҫбһҷбһӮбҹ„бһҡбһ–бһҸбһ¶бһҳбһҖбһҳбҹ’бһҡбһ·бһҸбһҮбҹҶбһ“бһ¶бһүбһҒбһ»бһҹбһӮбҹ’бһ“бһ¶
4. **бһҖбһ¶бһҡбһҡбҹҖбһ”бһ…бҹҶбһ…бһҳбҹ’бһӣбһҫбһҷ**бҹ– бһўбһ“бһ»бһңбһҸбҹ’бһҸбһҒбҹ’бһӣбһ№бһҳбһҹбһ¶бһҡбһҸбҹҶбһҡбһјбһңбһҡбһ…бһ“бһ¶бһҖбһ¶бһҡбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҹбҹҶбһҺбһҪбһҡбһ”бҹ’бһҡбһЎбһ„

### бһҮбҹҶбһ бһ¶бһ“бһўбһ“бһ»бһңбһҸбҹ’бһҸ

**бһҮбҹҶбһ бһ¶бһ“ бҹЎбҹ– бһҡбҹҖбһ”бһ…бҹҶбһ”бһҡбһ·бһҷбһ¶бһҖбһ¶бһҹбһҡбһ”бһҹбҹӢбһўбҹ’бһ“бһҖ**

```bash
pip install langchain langchain-openai
```

**бһҮбҹҶбһ бһ¶бһ“ бҹўбҹ– бһҳбһ»бһҒбһ„бһ¶бһҡбһҹбһ“бҹ’бһ‘бһ“бһ¶бһҳбһјбһӣбһҠбҹ’бһӢбһ¶бһ“**
- бһ”бһ„бҹ’бһҖбһҫбһҸбһҗбҹ’бһ“бһ¶бһҖбҹӢ `StudyAssistant`
- бһўбһ“бһ»бһңбһҸбҹ’бһҸбһўбһ“бһ»бһҹбҹ’бһҹбһ¶бһңбһҡбһёбһҷбҹҚбһҹбһ“бҹ’бһ‘бһ“бһ¶
- бһ”бһ“бҹ’бһҗбҹӮбһҳбһҖбһ¶бһҡбһҖбҹҶбһҺбһҸбҹӢбһ”бһ»бһӮбҹ’бһӮбһӣбһ·бһҖбһӣбһҖбҹ’бһҒбһҺбҹҲбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһӮбһ¶бҹҶбһ‘бҹ’бһҡбһўбһ”бҹӢбһҡбҹҶ

**бһҮбҹҶбһ бһ¶бһ“ бҹЈбҹ– бһ”бһ“бҹ’бһҗбҹӮбһҳбһ§бһ”бһҖбһҡбһҺбҹҚбһўбһ”бҹӢбһҡбҹҶ**
- **Code Explainer**бҹ– бһ–бһ“бҹ’бһҷбһӣбҹӢбһҖбһјбһҠбһҮбһ¶бһ•бҹ’бһ“бҹӮбһҖбһ„бһ¶бһҷбһҷбһӣбҹӢ
- **Quiz Generator**бҹ– бһ”бһ„бҹ’бһҖбһҫбһҸбһҹбҹҶбһҺбһҪбһҡбһўбҹҶбһ–бһёбһҳбҹҒбһҡбҹҖбһ“бһҖбһҳбҹ’бһҳбһңбһ·бһ’бһё
- **Progress Tracker**бҹ– бһҸбһ¶бһҳбһҠбһ¶бһ“бһ”бҹ’бһҡбһ’бһ¶бһ“бһ”бһ‘бһҠбҹӮбһӣбһ”бһ¶бһ“бһӮбҹ’бһҡбһ”бһҠбһҺбҹ’бһҸбһ”бҹӢ

**бһҮбҹҶбһ бһ¶бһ“ бҹӨбҹ– бһӣбһҖбҹ’бһҒбһҺбҹҲбһ–бһ·бһҹбҹҒбһҹбһҹбһҳбҹ’бһӣбҹҒбһ„бһ…бһ»бһ„бһҖбҹ’бһҡбҹ„бһҷ (бһҮбһ¶бһҮбһҳбҹ’бһҡбһҫбһҹ)**

- бһҸбҹ’бһҡбҹҖбһҳбһҒбҹ’бһӣбһҪбһ“бһҹбҹ’бһңбҹӮбһ„бһҷбһӣбҹӢбһҡбһ…бһ“бһ¶бһҹбһҳбҹ’бһ–бҹҗбһ“бҹ’бһ’ AI бһЎбһҫбһ„бһҮбһ¶бһ“бҹӢбһҒбҹ’бһ–бһҹбҹӢ (бһ—бҹ’бһ“бһ¶бһҖбҹӢбһ„бһ¶бһҡ бһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’бһ—бҹ’бһ“бһ¶бһҖбҹӢбһ„бһ¶бһҡбһ…бҹ’бһҡбһҫбһ“)

- бһҳбһ¶бһ“бһҹбһ„бҹ’бһғбһ№бһҳбһ”бһ„бҹ’бһҖбһҫбһҸбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһё AI бһ…бһҳбҹ’бһҡбһ»бҹҮбһҳбҹҒбһҢбҹҖ
- бһҳбһјбһӣбһҠбҹ’бһӢбһ¶бһ“бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһ”бһ„бҹ’бһҖбһҫбһҸ бһ“бһ·бһ„бһўбһ—бһ·бһңбһҢбҹ’бһҚбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһё AI бһ„бһ¶бһҷбһҹбҹ’бһҡбһҪбһӣбһҖбҹӮбһ„бһҸбһҳбҹ’бһҡбһ»бһҷ

## бһҹбһ„бҹ’бһҒбҹҒбһ”

рҹҺү бһўбҹ’бһ“бһҖбһ”бһ¶бһ“бһ‘бһ‘бһҪбһӣбһҮбҹҶбһ“бһ¶бһүбһҳбһјбһӣбһҠбҹ’бһӢбһ¶бһ“бһҖбҹ’бһ“бһ»бһ„бһҖбһ¶бһҡбһўбһ—бһ·бһңбһҢбҹ’бһҚбһ”бҹ’бһӣбһ»бһҖ AI бһ бһҫбһҷбһҡбҹҖбһ“бһ”бһ„бҹ’бһҖбһҫбһҸбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһё AI бһҹбҹ’бһҳбһ»бһӮбһҹбҹ’бһҳбһ¶бһүбһҠбҹ„бһҷбһ”бҹ’бһҡбһҫ LangChainбҹ” бһҠбһјбһ…бһҮбһ¶бһ”бһүбҹ’бһ…бһ”бҹӢбһҖбһ¶бһҡбһ”бһҺбҹ’бһҸбһ»бҹҮбһ”бһҺбҹ’бһҸбһ¶бһӣбһ–бһёбһҡбҹҖбһ”бһ…бҹҶбһӣбҹ’бһў бһўбҹ’бһ“бһҖбһ”бһ¶бһ“бһ‘бһ‘бһҪбһӣбһ§бһ”бһҖбһҡбһҺбҹҚбһҮбһ¶бһ…бҹ’бһҡбһҫбһ“бһ“бҹ…бһҖбҹ’бһ“бһ»бһ„бһҠбҹғбһҡбһ”бһҹбҹӢбһўбҹ’бһ“бһҖбҹ” бһҷбһҫбһ„бһҳбһҖбһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһўбҹ’бһңбһёбһҠбҹӮбһӣбһўбҹ’бһ“бһҖбһ”бһ¶бһ“бһҹбһҳбҹ’бһҡбҹҒбһ…бҹ”

### бһўбҹ’бһңбһёбһҠбҹӮбһӣбһўбҹ’бһ“бһҖбһ”бһ¶бһ“бһҡбҹҖбһ“

**бһӮбҹҶбһ“бһ·бһҸбһҳбһјбһӣбһҠбҹ’бһӢбһ¶бһ“бһ”бҹ’бһӣбһ»бһҖбҹ–**
- **бһўбһҸбҹ’бһҗбһ”бҹ’бһҡбһҷбҹ„бһҮбһ“бҹҚбһ”бҹ’бһӣбһ»бһҖ**бҹ– бһҷбһӣбҹӢбһ–бһёбһ–бҹҒбһӣбһҺбһ¶бһӮбһҪбһҡбһҮбҹ’бһҡбһҫбһҹбһҡбһҫбһҹбһ”бҹ’бһӣбһ»бһҖ бһ”бҹ’бһҡбһҹбһ·бһ“бһ”бһҫбһ’бҹҖбһ”бһ‘бҹ…бһ“бһ№бһ„бһҖбһ¶бһҡбһ бҹ… API бһҸбҹ’бһҡбһ„бҹӢ
- **бһҳбһјбһӣбһҠбҹ’бһӢбһ¶бһ“ LangChain**бҹ– бһҖбһ¶бһҡбһҡбҹҖбһ”бһ…бҹҶ бһ“бһ·бһ„бһҖбҹҶбһҺбһҸбҹӢбһҖбһ¶бһҡбһҸбһ—бҹ’бһҮбһ¶бһ”бҹӢбһҳбҹүбһјбһҢбҹӮбһӣ AI
- **бһ”бҹ’бһҡбһ—бҹҒбһ‘бһҹбһ¶бһҡ**бҹ– бһ”бҹ’бһҡбһҫ `SystemMessage`, `HumanMessage`, бһ“бһ·бһ„ `AIMessage` бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҖбһ¶бһҡбһҹбһ“бҹ’бһ‘бһ“бһ¶бһҹбҹҶбһҡбһ…бҹӢ

**бһӣбһҖбҹ’бһҒбһҺбҹҲбһ–бһ·бһҹбҹҒбһҹбһҖбһҳбҹ’бһҡбһ·бһҸбһҒбҹ’бһ–бһҹбҹӢбҹ–**
- **бһҖбһ¶бһҡбһ бҹ…бһ§бһ”бһҖбһҡбһҺбҹҚ**бҹ– бһ”бһ„бҹ’бһҖбһҫбһҸбһ“бһ·бһ„бһ”бһүбҹ’бһ…бһјбһӣбһ§бһ”бһҖбһҡбһҺбҹҚбһ”бһ“бҹ’бһҗбҹӮбһҳбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҳбһ»бһҒбһ„бһ¶бһҡ AI бһҒбҹ’бһ–бһҹбҹӢ
- **бһўбһ“бһ»бһҹбҹ’бһҹбһ¶бһңбһҡбһёбһҷбҹҚбһҹбһ“бҹ’бһ‘бһ“бһ¶**бҹ– бһҡбһҖбҹ’бһҹбһ¶бһ”бһҡбһ·бһ”бһ‘бһҹбһ“бҹ’бһ‘бһ“бһ¶бһ…бҹ’бһҡбһҫбһ“бһҮбһҪбһҡ
- **бһ…бһҳбҹ’бһӣбһҫбһҷбһ…бһҡбһ“бҹ’бһҸ**бҹ– бһўбһ“бһ»бһңбһҸбҹ’бһҸбһҖбһ¶бһҡбһ•бҹ’бһҸбһӣбҹӢбһ…бһҳбҹ’бһӣбһҫбһҷбһ–бҹҒбһӣбһңбҹҒбһӣбһ¶бһ–бһ·бһҸ
- **бһӮбҹҶбһҡбһјбһ”бһүбҹ’бһҮбһ¶**бҹ– бһ”бһ„бҹ’бһҖбһҫбһҸбһ–бһ¶бһҖбҹ’бһҷбһҮбҹҶбһ бһ¶бһ“бһҠбҹӮбһӣбһўбһ¶бһ…бһ”бҹ’бһҡбһҫбһЎбһҫбһ„бһңбһ·бһү
- **бһ…бҹҒбһүбһҳбһ»бһҒбһ„бһ¶бһҡбһҸбһ¶бһҳбһҡбһ…бһ“бһ¶бһҹбһҳбҹ’бһ–бҹҗбһ“бҹ’бһ’**бҹ– бһ”бһ„бҹ’бһҖбһҫбһҸбһ…бһҳбҹ’бһӣбһҫбһҷ AI бһҠбҹӮбһӣбһўбһ¶бһ…бһңбһ¶бһҷбһҸбһҳбҹ’бһӣбҹғбһ”бһ¶бһ“
- **Embeddings**бҹ– бһ”бһ„бҹ’бһҖбһҫбһҸбһҹбҹ’бһңбҹӮбһ„бһҡбһҖбһҹбҹҶбһ”бһјбһҡбһ”бҹӮбһ”бһўбһҸбҹ’бһҗбһ“бҹҗбһҷ бһ“бһ·бһ„бһҠбҹҶбһҺбһҫбһҡбһҖбһ¶бһҡбһҜбһҖбһҹбһ¶бһҡ

**бһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһҮбһ¶бһҖбҹӢбһҹбҹ’бһҸбҹӮбһ„бҹ–**
- **бһҖбһ¶бһҡбһҹбһ¶бһ„бһҹбһ„бҹӢбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһ–бҹҒбһүбһӣбҹҒбһү**бҹ– бһ”бһүбҹ’бһ…бһјбһӣбһҳбһ»бһҒбһ„бһ¶бһҡбһ…бҹ’бһҡбһҫбһ“бһ‘бҹ…бһҖбҹ’бһ“бһ»бһ„бһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһҡбһҪбһ…бһҮбһ¶бһҹбҹ’бһҡбҹҒбһ…
- **бһҖбһ¶бһҡбһҠбҹ„бҹҮбһҹбҹ’бһҡбһ¶бһҷбһҖбҹҶбһ бһ»бһҹ**бҹ– бһўбһ“бһ»бһңбһҸбҹ’бһҸбһҖбһ¶бһҡбһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„бһҖбҹҶбһ бһ»бһҹбһҷбҹүбһ¶бһ„бһҡбһ№бһ„бһҳбһ¶бҹҶ
- **бһҖбһ¶бһҡбһҡбһҪбһҳбһ”бһүбҹ’бһ…бһјбһӣбһ§бһ”бһҖбһҡбһҺбҹҚ**бҹ– бһ”бһ„бҹ’бһҖбһҫбһҸбһ§бһ”бһҖбһҡбһҺбҹҚбһ•бҹ’бһ‘бһ¶бһӣбҹӢбһҒбҹ’бһӣбһҪбһ“бһҠбҹӮбһӣбһ–бһ„бҹ’бһҡбһёбһҖбһҹбһҳбһҸбҹ’бһҗбһ—бһ¶бһ– AI

- бһ”бһ„бҹ’бһҖбһҫбһҸбһ§бһ”бһҖбһҡбһҺбҹҚбһҠбҹ„бҹҮбһҹбҹ’бһҡбһ¶бһҷбһ”бһүбҹ’бһ бһ¶бһҠбҹӮбһӣбһўбҹ’бһ“бһҖбһ”бҹ’бһҡбһҲбһҳбһ“бҹ…бһҖбҹ’бһ“бһ»бһ„бһҖбһ¶бһҡбһ„бһ¶бһҡ бһ¬бһӮбһҳбҹ’бһҡбҹ„бһ„бһҡбһ”бһҹбҹӢбһўбҹ’бһ“бһҖ

**бһҡбҹҖбһ”бһ…бҹҶбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҮбһ¶бһ“бҹӢбһҒбҹ’бһ–бһҹбҹӢбһ”бһ“бҹ’бһ‘бһ¶бһ”бҹӢ?**
- **бһ—бҹ’бһ“бһ¶бһҖбҹӢбһ„бһ¶бһҡ AI**бҹ– бһ”бһ„бҹ’бһҖбһҫбһҸбһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’ AI бһҠбҹӮбһӣбһўбһ¶бһ…бһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„бһ“бһ·бһ„бһўбһ“бһ»бһңбһҸбҹ’бһҸбһҖбһ·бһ…бҹ’бһ…бһҖбһ¶бһҡбһҹбҹ’бһҳбһ»бһӮбһҹбҹ’бһҳбһ¶бһүбһҠбҹ„бһҷбһҒбҹ’бһӣбһҪбһ“бһҜбһ„

- **AI бһ…бһҳбҹ’бһҡбһ»бҹҮбһҳбҹҒбһҢбҹҖ**бҹ– бһ”бҹ’бһҡбһҫбһўбһҸбҹ’бһҗбһ”бһ‘ бһҡбһјбһ”бһ—бһ¶бһ– бһ“бһ·бһ„бһҹбҹҶбһЎбҹҒбһ„бһҡбһҪбһҳбһӮбҹ’бһ“бһ¶ вҖ” бһңбһ·бһ’бһёбһҹбһ¶бһҹбҹ’бһҸбҹ’бһҡбһҮбһ¶бһ…бҹ’бһҡбһҫбһ“!
- **бһҮбһјбһ“бһҠбҹҶбһҺбһ№бһ„бһ•бһӣбһ·бһҸбһҖбһҳбҹ’бһҳ**бҹ– бһҡбҹҖбһ“бһ’бҹ’бһңбһҫбһұбҹ’бһҷбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһё AI бһҡбһ”бһҹбҹӢбһўбҹ’бһ“бһҖбһӣбҹҒбһ…бһ’бҹ’бһӣбҹ„ бһ“бһ·бһ„бһҸбһ¶бһҳбһҠбһ¶бһ“бһңбһ¶ бһ“бҹ…бһ–бһ·бһ—бһ–бһ–бһ·бһҸ

бһҘбһЎбһјбһңбһ“бҹҒбҹҮбһўбҹ’бһ“бһҖбһҳбһ¶бһ“бһ…бҹҶбһҺбҹҒбҹҮбһҠбһ№бһ„бһҖбҹ’бһ“бһ»бһ„бһҖбһ¶бһҡбһ”бһ„бҹ’бһҖбһҫбһҸбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһҹбһ“бҹ’бһ‘бһ“бһ¶бһҶбҹ’бһӣбһ¶бһҸбһңбҹғбһҠбҹӮбһӣбһўбһ¶бһ…бһҮбһҪбһҷбһҳбһ“бһ»бһҹбҹ’бһҹбһҠбҹ„бҹҮбһҹбҹ’бһҡбһ¶бһҷбһ”бһүбҹ’бһ бһ¶бһ–бһ·бһҸбҹ” бһҠбһјбһ…бһҮбһ¶бһҖбҹ’бһҡбһ»бһҳбһ бҹҠбһ»бһ“бһ”бһ»бһҡбһ¶бһҺбһҠбҹӮбһӣбһҡбһҪбһҳбһ”бһүбҹ’бһ…бһјбһӣбһ…бһҖбҹ’бһҒбһ»бһңбһ·бһҹбҹҗбһҷбһҡбһјбһ”бһҹбһ·бһӣбҹ’бһ”бҹҲбһҮбһ¶бһҳбһҪбһҷбһҮбҹҶбһ“бһ¶бһүбһ”бһ…бҹ’бһ…бҹҒбһҖбһ‘бҹҒбһҹ бһўбҹ’бһ“бһҖбһўбһ¶бһ…бһҹбһҳбҹ’бһҡбһ»бһҖбһҹбһҳбҹ’бһҡбһҪбһӣбһҹбһҳбһҸбҹ’бһҗбһ—бһ¶бһ– AI бһҮбһ¶бһҳбһҪбһҷбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһ”бҹ’бһҡбһҫбһ”бҹ’бһҡбһ¶бһҹбҹӢбһҮбһ¶бһҖбҹӢбһҹбҹ’бһҸбҹӮбһ„бҹ” бһҹбҹҶбһҺбһҪбһҡбһӮбһәбҹ– бһҸбһҫбһўбҹ’бһ“бһҖбһ“бһ№бһ„бһ”бһ„бҹ’бһҖбһҫбһҸбһўбҹ’бһңбһё? рҹҡҖ

## бһҖбһ¶бһҡбһ”бҹ’бһҡбһҖбһҪбһҸ GitHub Copilot Agent рҹҡҖ

бһ”бҹ’бһҡбһҫбһҡбһ”бҹҖбһ” Agent бһҠбһҫбһҳбҹ’бһ”бһёбһ”бһүбҹ’бһ…бһ”бҹӢбһ”бһүбҹ’бһ бһ¶бһҠбһјбһ…бһҒбһ¶бһ„бһҖбҹ’бһҡбҹ„бһҳбҹ–
